207 uint8_t SPS30Component::sht_crc_(uint8_t data1, uint8_t data2) {
208  uint8_t bit;
209  uint8_t crc = 0xFF;
210 
211  crc ^= data1;
212  for (bit = 8; bit > 0; --bit) {
213  if (crc & 0x80)
214  crc = (crc << 1) ^ 0x131;
215  else
216  crc = (crc << 1);

221  if (crc & 0x80)

243 bool SPS30Component::read_data_(uint16_t *data, uint8_t len) {
244  const uint8_t num_bytes = len * 3;
245  auto *buf = new uint8_t[num_bytes];
246 
247  if (!this->parent_->raw_receive(this->address_, buf, num_bytes)) {
248  delete[](buf);
249  return false;
250  }
251 
252  for (uint8_t i = 0; i < len; i++) {
253  const uint8_t j = 3 * i;
254  uint8_t crc = sht_crc_(buf[j], buf[j + 1]);
255  if (crc != buf[j + 2]) {
256  ESP_LOGE(TAG, "CRC8 Checksum invalid! 0x%02X != 0x%02X", buf[j + 2], crc);
257  delete[](buf);
258  return false;
259  }
260  data[i] = (buf[j] << 8) | buf[j + 1];
261  }
262 
263  delete[](buf);
264  return true;
