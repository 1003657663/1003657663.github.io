271  // Only print state when it's being changed

273  if (this->state_.value_or(current_state) != current_state) {
274  ESP_LOGD(TAG, " State: %s", ONOFF(v.is_on()));

276

279  }
280

282  ESP_LOGD(TAG, " Color Temperature: %.1f mireds", v.get_color_temperature());
283  }
284 
285  if (this->red_.has_value() || this->green_.has_value() || this->blue_.has_value()) {
286  ESP_LOGD(TAG, " Red=%.0f%%, Green=%.0f%%, Blue=%.0f%%", v.get_red() * 100.0f, v.get_green() * 100.0f,
287  v.get_blue() * 100.0f);
288  }
289  if (this->white_.has_value()) {
290  ESP_LOGD(TAG, " White Value: %.0f%%", v.get_white() * 100.0f);
291  }
292  }
293 
294  if (this->has_flash_()) {
295  // FLASH
296  if (this->publish_) {
297  ESP_LOGD(TAG, " Flash Length: %.1fs", *this->flash_length_ / 1e3f);
298  }
299

301  } else if (this->has_transition_()) {
302  // TRANSITION
303  if (this->publish_) {
304  ESP_LOGD(TAG, " Transition Length: %.1fs", *this->transition_length_ / 1e3f);
305  }
306 
307  // Special case: Transition and effect can be set when turning off
308  if (this->has_effect_()) {
309  if (this->publish_) {
310  ESP_LOGD(TAG, " Effect: 'None'");
311  }
312  this->parent_->stop_effect_();
313  }
314 
315  this->parent_->start_transition_(v, *this->transition_length_);
316 
317  } else if (this->has_effect_()) {
318  // EFFECT
319  auto effect = this->effect_;
320  const char *effect_s;
321  if (effect == 0)
322  effect_s = "None";
323  else
324  effect_s = this->parent_->effects_[*this->effect_ - 1]->get_name().c_str();
325 
326  if (this->publish_) {
327  ESP_LOGD(TAG, " Effect: '%s'", effect_s);
328  }
329 
330  this->parent_->start_effect_(*this->effect_);
331 
332  // Also set light color values when starting an effect
333  // For example to turn off the light
334  this->parent_->set_immediately_(v, true);
335  } else {
336  // INSTANT CHANGE
337  this->parent_->set_immediately_(v, this->publish_);
338  }
339 
340  if (!this->has_transition_()) {
341  this->parent_->target_state_reached_callback_.call();
342  }
343  if (this->publish_) {
344  this->parent_->publish_state();
345  }
346 
347  if (this->save_) {
348  LightStateRTCState saved;
349  saved.state = v.is_on();
350  saved.brightness = v.get_brightness();
351  saved.red = v.get_red();
352  saved.green = v.get_green();
353  saved.blue = v.get_blue();
354  saved.white = v.get_white();
355  saved.color_temp = v.get_color_temperature();
356  saved.effect = this->parent_->active_effect_index_;

361 LightColorValues LightCall::validate_() {
362  // use remote values for fallback
363  auto *name = this->parent_->get_name().c_str();
364  auto traits = this->parent_->get_traits();
365 
366  // Brightness exists check
367  if (this->brightness_.has_value() && !traits.get_supports_brightness()) {
368  ESP_LOGW(TAG, "'%s' - This light does not support setting brightness!", name);
369  this->brightness_.reset();
370  }
371 
372  // Transition length possible check
373  if (this->transition_length_.has_value() && *this->transition_length_ != 0 && !traits.get_supports_brightness()) {
374  ESP_LOGW(TAG, "'%s' - This light does not support transitions!", name);
375  this->transition_length_.reset();
376  }
377 
378  // RGB exists check
379  if (this->red_.has_value() || this->green_.has_value() || this->blue_.has_value()) {
380  if (!traits.get_supports_rgb()) {
381  ESP_LOGW(TAG, "'%s' - This light does not support setting RGB color!", name);
382  this->red_.reset();
383  this->green_.reset();
384  this->blue_.reset();
385  }
386  }
387 
388  // White value exists check
389  if (this->white_.has_value() && !traits.get_supports_rgb_white_value()) {
390  ESP_LOGW(TAG, "'%s' - This light does not support setting white value!", name);
391  this->white_.reset();
392  }
393 
394  // Color temperature exists check
395  if (this->color_temperature_.has_value() && !traits.get_supports_color_temperature()) {
396  ESP_LOGW(TAG, "'%s' - This light does not support setting color temperature!", name);
397  this->color_temperature_.reset();
398  }
399 
400  // sets RGB to 100% if only White specified
401  if (this->white_.has_value()) {
402  if (traits.get_supports_color_interlock()) {
403  if (!this->red_.has_value() && !this->green_.has_value() && !this->blue_.has_value()) {
404  this->red_ = optional<float>(1.0f);
405  this->green_ = optional<float>(1.0f);
406  this->blue_ = optional<float>(1.0f);
407  }
408  // make white values binary aka 0.0f or 1.0f...this allows brightness to do its job
409  if (*this->white_ > 0.0f) {
410  this->white_ = optional<float>(1.0f);
411  } else {
412  this->white_ = optional<float>(0.0f);
413  }
414  }
415  }
416  // White to 0% if (exclusively) setting any RGB value that isn't 255,255,255
417  else if (this->red_.has_value() || this->green_.has_value() || this->blue_.has_value()) {
418  if (traits.get_supports_color_interlock()) {
419  if (*this->red_ == 1.0f && *this->green_ == 1.0f && *this->blue_ == 1.0f &&
420  traits.get_supports_rgb_white_value() && traits.get_supports_color_interlock()) {
421  this->white_ = optional<float>(1.0f);
422  } else if (!this->white_.has_value() || !traits.get_supports_rgb_white_value()) {
423  this->white_ = optional<float>(0.0f);
424  }
425  }
426  }
427  // if changing Kelvin alone, change to white light
428  else if (this->color_temperature_.has_value()) {
429  if (!traits.get_supports_color_interlock()) {
430  if (!this->red_.has_value() && !this->green_.has_value() && !this->blue_.has_value()) {
431  this->red_ = optional<float>(1.0f);
432  this->green_ = optional<float>(1.0f);
433  this->blue_ = optional<float>(1.0f);
434  }
435  }
436  // if setting Kelvin from color (i.e. switching to white light), set White to 100%
437  auto cv = this->parent_->remote_values;
438  bool was_color = cv.get_red() != 1.0f || cv.get_blue() != 1.0f || cv.get_green() != 1.0f;
439  bool now_white = *this->red_ == 1.0f && *this->blue_ == 1.0f && *this->green_ == 1.0f;
440  if (traits.get_supports_color_interlock()) {
441  if (cv.get_white() < 1.0f) {
442  this->white_ = optional<float>(1.0f);
443  }
444 
445  if (was_color && !this->red_.has_value() && !this->green_.has_value() && !this->blue_.has_value()) {
446  this->red_ = optional<float>(1.0f);
447  this->green_ = optional<float>(1.0f);
448  this->blue_ = optional<float>(1.0f);
449  }
450  } else {
451  if (!this->white_.has_value() && was_color && now_white) {
452  this->white_ = optional<float>(1.0f);
453  }
454  }
455  }

497  this->flash_length_.reset();

528  this->transition_length_ = this->parent_->default_transition_length_;
529  }

533  this->transition_length_.reset();
534  }

542  // Do not use light color values directly, so that effects can set 0% brightness
