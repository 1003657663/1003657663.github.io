305 size_t OTAComponent::wait_receive_(uint8_t *buf, size_t bytes, bool check_disconnected) {
306  size_t available = 0;
307  uint32_t start = millis();
308  do {
309  App.feed_wdt();
310  if (check_disconnected && !this->client_.connected()) {
311  ESP_LOGW(TAG, "Error client disconnected while receiving data!");
312  return 0;
313  }
314  int availi = this->client_.available();
315  if (availi < 0) {
316  ESP_LOGW(TAG, "Error reading data!");
317  return 0;
318  }
319  uint32_t now = millis();
320  if (availi == 0 && now - start > 10000) {
321  ESP_LOGW(TAG, "Timeout waiting for data!");
322  return 0;
323  }
324  available = size_t(availi);
325  yield();
326  } while (bytes == 0 ? available == 0 : available < bytes);
327 
328  if (bytes == 0)
329  bytes = std::min(available, size_t(1024));
330 
331  bool success = false;
332  for (uint32_t i = 0; !success && i < 100; i++) {
333  int res = this->client_.read(buf, bytes);
334 
335  if (res != int(bytes)) {
336  // ESP32 implementation has an issue where calling read can fail with EAGAIN (race condition)
337  // so just re-try it until it works (with generous timeout of 1s)
338  // because we check with available() first this should not cause us any trouble in all other cases
339  delay(10);
340  } else {
341  success = true;
342  }
343  }
344 
345  if (!success) {
346  ESP_LOGW(TAG, "Reading %u bytes of binary data failed!", bytes); // NOLINT
347  return 0;
348  }
349 
350  return bytes;
351 }
