10 inline static int16_t sin16_c(uint16_t theta) {
11  static const uint16_t BASE[] = {0, 6393, 12539, 18204, 23170, 27245, 30273, 32137};
12  static const uint8_t SLOPE[] = {49, 48, 44, 38, 31, 23, 14, 4};
13  uint16_t offset = (theta & 0x3FFF) >> 3; // 0..2047
14  if (theta & 0x4000)
15  offset = 2047 - offset;
16  uint8_t section = offset / 256; // 0..7
17  uint16_t b = BASE[section];
18  uint8_t m = SLOPE[section];
19  uint8_t secoffset8 = uint8_t(offset) / 2;
20  uint16_t mx = m * secoffset8;
21  int16_t y = mx + b;
22  if (theta & 0x8000)
23  return -y;
24  return y;
25 }
26 inline static uint8_t half_sin8(uint8_t v) { return sin16_c(uint16_t(v) * 128u) >> 8; }
