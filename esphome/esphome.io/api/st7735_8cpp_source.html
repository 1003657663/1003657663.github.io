349 void ST7735::display_init_(const uint8_t *addr) {
350  uint8_t num_commands, cmd, num_args;
351  uint16_t ms;
352 
353  num_commands = pgm_read_byte(addr++); // Number of commands to follow
354  while (num_commands--) { // For each command...
355  cmd = pgm_read_byte(addr++); // Read command
356  num_args = pgm_read_byte(addr++); // Number of args to follow
357  ms = num_args & ST_CMD_DELAY; // If hibit set, delay follows args
358  num_args &= ~ST_CMD_DELAY; // Mask out delay bit
359  this->sendcommand_(cmd, addr, num_args);
360  addr += num_args;
361 
362  if (ms) {
363  ms = pgm_read_byte(addr++); // Read post-command delay time (ms)
364  if (ms == 255)
365  ms = 500; // If 255, delay for 500 ms
366  delay(ms);
367  }
368  }
