127 void YashimaClimate::transmit_state_() {
128  uint8_t remote_state[YASHIMA_STATE_LENGTH] = {0};
129 
130  remote_state[0] = YASHIMA_BASE_BYTE0;
131  remote_state[1] = YASHIMA_BASE_BYTE1;
132  remote_state[2] = YASHIMA_BASE_BYTE2;
133  remote_state[3] = YASHIMA_BASE_BYTE3;
134  remote_state[4] = YASHIMA_BASE_BYTE4;
135  remote_state[5] = YASHIMA_BASE_BYTE5;
136  remote_state[6] = YASHIMA_BASE_BYTE6;
137  remote_state[7] = YASHIMA_BASE_BYTE7;
138  remote_state[8] = YASHIMA_BASE_BYTE8;
139 
140  // Set mode
141  switch (this->mode) {
142  case climate::CLIMATE_MODE_AUTO:
143  remote_state[0] |= YASHIMA_MODE_AUTO_BYTE0;
144  remote_state[5] |= YASHIMA_MODE_AUTO_BYTE5;
145  break;
146  case climate::CLIMATE_MODE_COOL:
147  remote_state[0] |= YASHIMA_MODE_COOL_BYTE0;
148  remote_state[5] |= YASHIMA_MODE_COOL_BYTE5;
149  break;
150  case climate::CLIMATE_MODE_HEAT:
151  remote_state[0] |= YASHIMA_MODE_HEAT_BYTE0;
152  remote_state[5] |= YASHIMA_MODE_HEAT_BYTE5;
153  break;
154  case climate::CLIMATE_MODE_OFF:
155  default:
156  remote_state[0] |= YASHIMA_MODE_OFF_BYTE0;
157  remote_state[5] |= YASHIMA_MODE_OFF_BYTE5;
158  break;
159  // TODO: CLIMATE_MODE_FAN_ONLY, CLIMATE_MODE_DRY are missing in esphome
160  }
161 
162  // TODO: missing support for fan speed
163  remote_state[2] |= YASHIMA_FAN_AUTO_BYTE2;
164 
165  // Set temperature
166  uint8_t safecelsius = std::max((uint8_t) this->target_temperature, YASHIMA_TEMP_MIN);
167  safecelsius = std::min(safecelsius, YASHIMA_TEMP_MAX);
168  remote_state[1] |= YASHIMA_TEMP_MAP_BYTE1[safecelsius - YASHIMA_TEMP_MIN];
169 
170  auto transmit = this->transmitter_->transmit();
171  auto data = transmit.get_data();
172 
173  data->set_carrier_frequency(YASHIMA_CARRIER_FREQUENCY);
174 
175  // Header
176  data->mark(YASHIMA_HEADER_MARK);
177  data->space(YASHIMA_HEADER_SPACE);
178  // Data (sent from the MSB to the LSB)
179  for (uint8_t i : remote_state)
180  for (int8_t j = 7; j >= 0; j--) {
181  data->mark(YASHIMA_BIT_MARK);
182  bool bit = i & (1 << j);
183  data->space(bit ? YASHIMA_ONE_SPACE : YASHIMA_ZERO_SPACE);
184  }
185  // Footer
186  data->mark(YASHIMA_BIT_MARK);
187  data->space(YASHIMA_GAP);
188 
189  transmit.perform();
190 }
