36 void AS3935Component::loop() {
37  if (!this->irq_pin_->digital_read())
38  return;
39 
40  uint8_t int_value = this->read_interrupt_register_();
41  if (int_value == NOISE_INT) {
42  ESP_LOGI(TAG, "Noise was detected - try increasing the noise level value!");
43  } else if (int_value == DISTURBER_INT) {
44  ESP_LOGI(TAG, "Disturber was detected - try increasing the spike rejection value!");
45  } else if (int_value == LIGHTNING_INT) {
46  ESP_LOGI(TAG, "Lightning has been detected!");
47  if (this->thunder_alert_binary_sensor_ != nullptr)
48  this->thunder_alert_binary_sensor_->publish_state(true);
49  uint8_t distance = this->get_distance_to_storm_();
50  if (this->distance_sensor_ != nullptr)
51  this->distance_sensor_->publish_state(distance);
52  uint32_t energy = this->get_lightning_energy_();
53  if (this->energy_sensor_ != nullptr)
54  this->energy_sensor_->publish_state(energy);
55  }
56  this->thunder_alert_binary_sensor_->publish_state(false);
57 }

166 // REG0x03, bits [3:0], manufacturer default: 0.
167 // When there is an event that exceeds the watchdog threshold, the register is written
168 // with the type of event. This consists of two messages: INT_D (disturber detected) and
169 // INT_L (Lightning detected). A third interrupt INT_NH (noise level too HIGH)
170 // indicates that the noise level has been exceeded and will persist until the
171 // noise has ended. Events are active HIGH. There is a one second window of time to
172 // read the interrupt register after lightning is detected, and 1.5 after
173 // disturber.
174 uint8_t AS3935Component::read_interrupt_register_() {
175  // A 2ms delay is added to allow for the memory register to be populated
176  // after the interrupt pin goes HIGH. See "Interrupt Management" in
177  // datasheet.
178  ESP_LOGV(TAG, "Calling read_interrupt_register_");
179  delay(2);
180  return this->read_register_(INT_MASK_ANT, INT_MASK);
181 }

202 uint32_t AS3935Component::get_lightning_energy_() {
203  ESP_LOGV(TAG, "Calling get_lightning_energy_");
204  uint32_t pure_light = 0; // Variable for lightning energy which is just a pure number.
205  uint32_t temp = 0;
206  // Temp variable for lightning energy.
207  temp = this->read_register_(ENERGY_LIGHT_MMSB, ENERGY_MASK);
208  // Temporary Value is large enough to handle a shift of 16 bits.
209  pure_light = temp << 16;
210  temp = this->read_register(ENERGY_LIGHT_MSB);
211  // Temporary value is large enough to handle a shift of 8 bits.
212  pure_light |= temp << 8;
213  // No shift here, directly OR'ed into pure_light variable.
214  temp = this->read_register(ENERGY_LIGHT_LSB);
215  pure_light |= temp;
216  return pure_light;
217 }
