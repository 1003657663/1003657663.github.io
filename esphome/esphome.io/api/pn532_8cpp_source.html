110 
111  uint8_t num_targets = read[0];
112  if (num_targets != 1) {
113  // no tags found or too many
114  this->current_uid_ = {};
115  this->turn_off_rf_();
116  return;
117  }
118 
119  uint8_t nfcid_length = read[5];
120  std::vector<uint8_t> nfcid(read.begin() + 6, read.begin() + 6 + nfcid_length);
121  if (read.size() < 6U + nfcid_length) {
122  // oops, pn532 returned invalid data
123  return;
124  }
125 
126  bool report = true;
127  for (auto *bin_sens : this->binary_sensors_) {
128  if (bin_sens->process(nfcid)) {
129  report = false;
130  }
131  }
132 
133  if (nfcid.size() == this->current_uid_.size()) {
134  bool same_uid = false;
135  for (uint8_t i = 0; i < nfcid.size(); i++)
136  same_uid |= nfcid[i] == this->current_uid_[i];
137  if (same_uid)
138  return;
139  }

141  this->current_uid_ = nfcid;
142

145  for (auto *trigger : this->triggers_)

147

156  }
157  }
158  }
159  } else if (next_task_ == CLEAN) {

161  if (!this->clean_tag_(nfcid)) {
162  ESP_LOGE(TAG, " Tag was not fully cleaned successfully");
163  }
164  ESP_LOGD(TAG, " Tag cleaned!");
165  } else if (next_task_ == FORMAT) {
166  ESP_LOGD(TAG, " Tag formatting...");
167  if (!this->format_tag_(nfcid)) {
168  ESP_LOGE(TAG, "Error formatting tag as NDEF");
169  }
170  ESP_LOGD(TAG, " Tag formatted!");
171  } else if (next_task_ == WRITE) {
172  if (this->next_task_message_to_write_ != nullptr) {
173  ESP_LOGD(TAG, " Tag writing...");
174  ESP_LOGD(TAG, " Tag formatting...");
175  if (!this->format_tag_(nfcid)) {
176  ESP_LOGE(TAG, " Tag could not be formatted for writing");

178  ESP_LOGD(TAG, " Writing NDEF data");

180  ESP_LOGE(TAG, " Failed to write message to tag");

186  }

191

195 bool PN532::write_command_(const std::vector<uint8_t> &data) {
196  std::vector<uint8_t> write_data;
197  // Preamble
198  write_data.push_back(0x00);
199 
200  // Start code
201  write_data.push_back(0x00);
202  write_data.push_back(0xFF);
203 
204  // Length of message, TFI + data bytes
205  const uint8_t real_length = data.size() + 1;
206  // LEN
207  write_data.push_back(real_length);
208  // LCS (Length checksum)
209  write_data.push_back(~real_length + 1);
210 
211  // TFI (Frame Identifier, 0xD4 means to PN532, 0xD5 means from PN532)
212  write_data.push_back(0xD4);
213  // calculate checksum, TFI is part of checksum
214  uint8_t checksum = 0xD4;
215 
216  // DATA
217  for (uint8_t dat : data) {
218  write_data.push_back(dat);
219  checksum += dat;
220  }
221 
222  // DCS (Data checksum)
223  write_data.push_back(~checksum + 1);
224  // Postamble
225  write_data.push_back(0x00);
226 
227  this->write_data(write_data);
228 
229  return this->read_ack_();
230 }
