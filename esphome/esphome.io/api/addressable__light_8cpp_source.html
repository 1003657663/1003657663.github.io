9 Color ESPHSVColor::to_rgb() const {
10  // based on FastLED's hsv rainbow to rgb
11  const uint8_t hue = this->hue;
12  const uint8_t sat = this->saturation;
13  const uint8_t val = this->value;
14  // upper 3 hue bits are for branch selection, lower 5 are for values
15  const uint8_t offset8 = (hue & 0x1F) << 3; // 0..248
16  // third of the offset, 255/3 = 85 (actually only up to 82; 164)
17  const uint8_t third = esp_scale8(offset8, 85);
18  const uint8_t two_thirds = esp_scale8(offset8, 170);
19  Color rgb(255, 255, 255, 0);
20  switch (hue >> 5) {
21  case 0b000:
22  rgb.r = 255 - third;
23  rgb.g = third;
24  rgb.b = 0;
25  break;
26  case 0b001:
27  rgb.r = 171;
28  rgb.g = 85 + third;
29  rgb.b = 0;
30  break;
31  case 0b010:
32  rgb.r = 171 - two_thirds;
33  rgb.g = 170 + third;
34  rgb.b = 0;
35  break;
36  case 0b011:
37  rgb.r = 0;
38  rgb.g = 255 - third;
39  rgb.b = third;
40  break;
41  case 0b100:
42  rgb.r = 0;
43  rgb.g = 171 - two_thirds;
44  rgb.b = 85 + two_thirds;
45  break;
46  case 0b101:
47  rgb.r = third;
48  rgb.g = 0;
49  rgb.b = 255 - third;
50  break;
51  case 0b110:
52  rgb.r = 85 + third;
53  rgb.g = 0;
54  rgb.b = 171 - third;
55  break;
56  case 0b111:
57  rgb.r = 170 + third;
58  rgb.g = 0;
59  rgb.b = 85 - third;
60  break;
61  default:
62  break;
63  }
64  // low saturation -> add uniform color to orig. hue
65  // high saturation -> use hue directly
66  // scales with square of saturation
67  // (r,g,b) = (r,g,b) * sat + (1 - sat)^2
68  rgb *= sat;
69  const uint8_t desat = 255 - sat;
70  rgb += esp_scale8(desat, desat);
71  // (r,g,b) = (r,g,b) * val
72  rgb *= val;
73  return rgb;
74 }

187 void AddressableLight::write_state(LightState *state) {
188  auto val = state->current_values;
189  auto max_brightness = static_cast<uint8_t>(roundf(val.get_brightness() * val.get_state() * 255.0f));
190  this->correction_.set_local_brightness(max_brightness);
191 
192  this->last_transition_progress_ = 0.0f;
193  this->accumulated_alpha_ = 0.0f;
194 
195  if (this->is_effect_active())
196  return;
197 
198  // don't use LightState helper, gamma correction+brightness is handled by ESPColorView
199 
200  if (state->transformer_ == nullptr || !state->transformer_->is_transition()) {
201  // no transformer active or non-transition one
202  this->all() = esp_color_from_light_color_values(val);
203  } else {
204  // transition transformer active, activate specialized transition for addressable effects
205  // instead of using a unified transition for all LEDs, we use the current state each LED as the
206  // start. Warning: ugly
207 
208  // We can't use a direct lerp smoothing here though - that would require creating a copy of the original
209  // state of each LED at the start of the transition
210  // Instead, we "fake" the look of the LERP by using an exponential average over time and using
211  // dynamically-calculated alpha values to match the look of the
212 
213  float new_progress = state->transformer_->get_progress();
214  float prev_smoothed = LightTransitionTransformer::smoothed_progress(last_transition_progress_);
215  float new_smoothed = LightTransitionTransformer::smoothed_progress(new_progress);
216  this->last_transition_progress_ = new_progress;
217 
218  auto end_values = state->transformer_->get_end_values();
219  Color target_color = esp_color_from_light_color_values(end_values);
220 
221  // our transition will handle brightness, disable brightness in correction.
222  this->correction_.set_local_brightness(255);
223  uint8_t orig_w = target_color.w;
224  target_color *= static_cast<uint8_t>(roundf(end_values.get_brightness() * end_values.get_state() * 255.0f));
225  // w is not scaled by brightness
226  target_color.w = orig_w;
227 
228  float denom = (1.0f - new_smoothed);
229  float alpha = denom == 0.0f ? 0.0f : (new_smoothed - prev_smoothed) / denom;
230 
231  // We need to use a low-resolution alpha here which makes the transition set in only after ~half of the length
232  // We solve this by accumulating the fractional part of the alpha over time.
233  float alpha255 = alpha * 255.0f;
234  float alpha255int = floorf(alpha255);
235  float alpha255remainder = alpha255 - alpha255int;
236 
237  this->accumulated_alpha_ += alpha255remainder;
238  float alpha_add = floorf(this->accumulated_alpha_);
239  this->accumulated_alpha_ -= alpha_add;
240 
241  alpha255 += alpha_add;
242  alpha255 = clamp(alpha255, 0.0f, 255.0f);
243  auto alpha8 = static_cast<uint8_t>(alpha255);
244 
245  if (alpha8 != 0) {
246  uint8_t inv_alpha8 = 255 - alpha8;
247  Color add = target_color * alpha8;
248 
249  for (auto led : *this)
250  led = add + led.get() * inv_alpha8;
251  }
252  }
253 
254  this->schedule_show();
255 }

257 void ESPColorCorrection::calculate_gamma_table(float gamma) {
258  for (uint16_t i = 0; i < 256; i++) {
259  // corrected = val ^ gamma
260  auto corrected = static_cast<uint8_t>(roundf(255.0f * gamma_correct(i / 255.0f, gamma)));
261  this->gamma_table_[i] = corrected;
262  }
263  if (gamma == 0.0f) {
264  for (uint16_t i = 0; i < 256; i++)
265  this->gamma_reverse_table_[i] = i;
266  return;
267  }
268  for (uint16_t i = 0; i < 256; i++) {
269  // val = corrected ^ (1/gamma)
270  auto uncorrected = static_cast<uint8_t>(roundf(255.0f * powf(i / 255.0f, 1.0f / gamma)));
271  this->gamma_reverse_table_[i] = uncorrected;
272  }
273 }
