179 float MAX31865Sensor::calc_temperature_(const float& rtd_ratio) {
180  // Based loosely on Adafruit's library: https://github.com/adafruit/Adafruit_MAX31865
181  // Mainly based on formulas provided by Analog:
182  // http://www.analog.com/media/en/technical-documentation/application-notes/AN709_0.pdf
183 
184  const float a = 3.9083e-3;
185  const float b = -5.775e-7;
186  const float z1 = -a;
187  const float z2 = a * a - 4 * b;
188  const float z3 = 4 * b / rtd_nominal_resistance_;
189  const float z4 = 2 * b;
190 
191  float rtd_resistance = rtd_ratio * reference_resistance_;
192 
193  // ≥ 0°C Formula
194  const float pos_temp = (z1 + std::sqrt(z2 + (z3 * rtd_resistance))) / z4;
195  if (pos_temp >= 0) {
196  return pos_temp;
197  }
198 
199  // < 0°C Formula
200  if (rtd_nominal_resistance_ != 100) {
201  // Normalize RTD to 100Ω
202  rtd_resistance /= rtd_nominal_resistance_;
203  rtd_resistance *= 100;
204  }
205  float rpoly = rtd_resistance;
206  float neg_temp = -242.02;
207  neg_temp += 2.2228 * rpoly;
208  rpoly *= rtd_resistance; // square
209  neg_temp += 2.5859e-3 * rpoly;
210  rpoly *= rtd_resistance; // ^3
211  neg_temp -= 4.8260e-6 * rpoly;
212  rpoly *= rtd_resistance; // ^4
213  neg_temp -= 2.8183e-8 * rpoly;
214  rpoly *= rtd_resistance; // ^5
215  neg_temp += 1.5243e-10 * rpoly;
216  return neg_temp;
217 }
