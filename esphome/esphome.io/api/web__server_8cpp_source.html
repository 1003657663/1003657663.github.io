41 UrlMatch match_url(const std::string &url, bool only_domain = false) {
42  UrlMatch match;
43  match.valid = false;
44  size_t domain_end = url.find('/', 1);
45  if (domain_end == std::string::npos)
46  return match;
47  match.domain = url.substr(1, domain_end - 1);
48  if (only_domain) {
49  match.valid = true;
50  return match;
51  }
52  if (url.length() == domain_end - 1)
53  return match;
54  size_t id_begin = domain_end + 1;
55  size_t id_end = url.find('/', id_begin);
56  match.valid = true;
57  if (id_end == std::string::npos) {
58  match.id = url.substr(id_begin, url.length() - id_begin);
59  return match;
60  }
61  match.id = url.substr(id_begin, id_end - id_begin);
62  size_t method_begin = id_end + 1;
63  match.method = url.substr(method_begin, url.length() - method_begin);
64  return match;
65 }
