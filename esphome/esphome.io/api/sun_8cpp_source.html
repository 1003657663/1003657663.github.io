63 double Sun::sun_declination_(double sun_time) {
64  double n = sun_time - 1.0;
65  // maximum declination
66  const double tot = -sin(EARTH_TILT);
67 
68  // eccentricity of the earth's orbit (ellipse)
69  double eccentricity = 0.0167;
70 
71  // days since perihelion (January 3rd)
72  double days_since_perihelion = n - 2;
73  // days since december solstice (december 22)
74  double days_since_december_solstice = n + 10;
75  const double c = TAU / 365.24;
76  double v = cos(c * days_since_december_solstice + 2 * eccentricity * sin(c * days_since_perihelion));
77  // Make sure value is in range (double error may lead to results slightly larger than 1)
78  double x = clampd(tot * v, -1.0, 1.0);

138 double Sun::sun_time_for_elevation_(int32_t day_of_year, double elevation, bool rising) {
139  // Use binary search, newton's method would be better but binary search already
140  // converges quite well (19 cycles) and much simpler. Function is guaranteed to be
141  // monotonous.
142  double lo, hi;
143  if (rising) {
144  lo = day_of_year + 0.0;
145  hi = day_of_year + 0.5;
146  } else {
147  lo = day_of_year + 1.0;
148  hi = day_of_year + 0.5;
149  }
150 
151  double min_elevation = this->elevation_(lo);
152  double max_elevation = this->elevation_(hi);
153  if (elevation < min_elevation || elevation > max_elevation)
154  return NAN;
155 
156  // Accuracy: 0.1s
157  const double accuracy = 1.0 / (24.0 * 60.0 * 60.0 * 10.0);
158 
159  while (fabs(hi - lo) > accuracy) {
160  double mid = (lo + hi) / 2.0;
161  double value = this->elevation_(mid) - elevation;
162  if (value < 0) {
163  lo = mid;
164  } else if (value > 0) {
165  hi = mid;
166  } else {
167  lo = hi = mid;
168  break;
169  }
170  }
171 
172  return (lo + hi) / 2.0;
173 }
