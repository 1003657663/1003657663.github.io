24 uint32_t UARTComponent::get_config() {
25  uint32_t config = 0;
26 
27  /*
28  * All bits numbers below come from
29  * framework-arduinoespressif32/cores/esp32/esp32-hal-uart.h
30  * And more specifically conf0 union in uart_dev_t.
31  *
32  * Below is bit used from conf0 union.
33  * <name>:<bits position> <values>
34  * parity:0 0:even 1:odd
35  * parity_en:1 Set this bit to enable uart parity check.
36  * bit_num:2-4 0:5bits 1:6bits 2:7bits 3:8bits
37  * stop_bit_num:4-6 stop bit. 1:1bit 2:1.5bits 3:2bits
38  * tick_ref_always_on:27 select the clock.1：apb clock：ref_tick
39  */
40 
41  if (this->parity_ == UART_CONFIG_PARITY_EVEN)
42  config |= UART_PARITY_EVEN | UART_PARITY_EN;
43  else if (this->parity_ == UART_CONFIG_PARITY_ODD)
44  config |= UART_PARITY_ODD | UART_PARITY_EN;
45 
46  switch (this->data_bits_) {
47  case 5:
48  config |= UART_NB_BIT_5;
49  break;
50  case 6:
51  config |= UART_NB_BIT_6;
52  break;
53  case 7:
54  config |= UART_NB_BIT_7;
55  break;
56  case 8:
57  config |= UART_NB_BIT_8;
58  break;
59  }
60 
61  if (this->stop_bits_ == 1)
62  config |= UART_NB_STOP_BIT_1;
63  else
64  config |= UART_NB_STOP_BIT_2;
65
