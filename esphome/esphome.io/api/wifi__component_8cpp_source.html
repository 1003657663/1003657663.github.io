329 void WiFiComponent::check_scanning_finished() {
330  if (!this->scan_done_) {
331  if (millis() - this->action_started_ > 30000) {
332  ESP_LOGE(TAG, "Scan timeout!");
333  this->retry_connect();
334  }
335  return;
336  }
337  this->scan_done_ = false;
338 
339  ESP_LOGD(TAG, "Found networks:");
340  if (this->scan_result_.empty()) {
341  ESP_LOGD(TAG, " No network found!");
342  this->retry_connect();
343  return;
344  }
345 
346  for (auto &res : this->scan_result_) {
347  for (auto &ap : this->sta_) {
348  if (res.matches(ap)) {
349  res.set_matches(true);
350  if (!this->has_sta_priority(res.get_bssid())) {
351  this->set_sta_priority(res.get_bssid(), ap.get_priority());
352  }
353  res.set_priority(this->get_sta_priority(res.get_bssid()));
354  break;
355  }
356  }
357  }
358 
359  std::stable_sort(this->scan_result_.begin(), this->scan_result_.end(),
360  [](const WiFiScanResult &a, const WiFiScanResult &b) {
361  // return true if a is better than b
362  if (a.get_matches() && !b.get_matches())
363  return true;
364  if (!a.get_matches() && b.get_matches())
365  return false;
366 
367  if (a.get_matches() && b.get_matches()) {
368  // if both match, check priority
369  if (a.get_priority() != b.get_priority())
370  return a.get_priority() > b.get_priority();
371  }
372 
373  return a.get_rssi() > b.get_rssi();
374  });
375 
376  for (auto &res : this->scan_result_) {
377  char bssid_s[18];
378  auto bssid = res.get_bssid();
379  sprintf(bssid_s, "%02X:%02X:%02X:%02X:%02X:%02X", bssid[0], bssid[1], bssid[2], bssid[3], bssid[4], bssid[5]);
380  char signal_bars[50];
381  print_signal_bars(res.get_rssi(), signal_bars);
382 
383  if (res.get_matches()) {
384  ESP_LOGI(TAG, "- '%s' %s" LOG_SECRET("(%s) ") "%s", res.get_ssid().c_str(),
385  res.get_is_hidden() ? "(HIDDEN) " : "", bssid_s, signal_bars);
386  ESP_LOGD(TAG, " Channel: %u", res.get_channel());
387  ESP_LOGD(TAG, " RSSI: %d dB", res.get_rssi());
388  } else {
389  ESP_LOGD(TAG, "- " LOG_SECRET("'%s'") " " LOG_SECRET("(%s) ") "%s", res.get_ssid().c_str(), bssid_s, signal_bars);
390  }
391  }
392 
393  if (!this->scan_result_[0].get_matches()) {
394  ESP_LOGW(TAG, "No matching network found!");
395  this->retry_connect();
396  return;
397  }
398 
399  WiFiAP connect_params;
400  WiFiScanResult scan_res = this->scan_result_[0];
401  for (auto &config : this->sta_) {
402  // search for matching STA config, at least one will match (from checks before)
403  if (!scan_res.matches(config)) {
404  continue;
405  }
406 
407  if (config.get_hidden()) {
408  // selected network is hidden, we use the data from the config
409  connect_params.set_hidden(true);
410  connect_params.set_ssid(config.get_ssid());
411  // don't set BSSID and channel, there might be multiple hidden networks
412  // but we can't know which one is the correct one. Rely on probe-req with just SSID.
413  } else {
414  // selected network is visible, we use the data from the scan
415  // limit the connect params to only connect to exactly this network
416  // (network selection is done during scan phase).
417  connect_params.set_hidden(false);
418  connect_params.set_ssid(scan_res.get_ssid());
419  connect_params.set_channel(scan_res.get_channel());
420  connect_params.set_bssid(scan_res.get_bssid());
421  }
422  // copy manual IP (if set)
423  connect_params.set_manual_ip(config.get_manual_ip());
424 
425 #ifdef ESPHOME_WIFI_WPA2_EAP
426  // copy EAP parameters (if set)
427  connect_params.set_eap(config.get_eap());
428 #endif
429 
430  // copy password (if set)
431  connect_params.set_password(config.get_password());
432 
433  break;
434  }
435 
436  yield();
437 
438  this->selected_ap_ = connect_params;
439  this->start_connecting(connect_params, false);
440 }

461 #ifdef USE_CAPTIVE_PORTAL
462  if (this->is_captive_portal_active_()) {

464  }
465 #endif
466  ESP_LOGD(TAG, "Disabling AP...");
467  this->wifi_mode_({}, false);
468  }

473  this->num_retried_ = 0;
474  return;

476 
477  uint32_t now = millis();
478  if (now - this->action_started_ > 30000) {
479  ESP_LOGW(TAG, "Timeout while connecting to WiFi.");
480  this->retry_connect();
481  return;
482  }
483 
484  if (this->error_from_callback_) {
485  ESP_LOGW(TAG, "Error while connecting to network.");
486  this->retry_connect();
487  return;
488  }
489 
490  if (status == WL_IDLE_STATUS || status == WL_DISCONNECTED || status == WL_CONNECTION_LOST) {
491  // WL_DISCONNECTED is set while not connected yet.
492  // WL_IDLE_STATUS is set while we're waiting for the IP address.
493  // WL_CONNECTION_LOST happens on the ESP32
494  return;
495  }
496 
497  if (status == WL_NO_SSID_AVAIL) {
498  ESP_LOGW(TAG, "WiFi network can not be found anymore.");
499  this->retry_connect();
500  return;
501  }
502 
503  if (status == WL_CONNECT_FAILED) {
504  ESP_LOGW(TAG, "Connecting to WiFi network failed. Are the credentials wrong?");
505  this->retry_connect();
506  return;
507  }
508 
509  ESP_LOGW(TAG, "WiFi Unknown connection status %d", status);
510 }

512 void WiFiComponent::retry_connect() {
513  if (this->selected_ap_.get_bssid()) {
514  auto bssid = *this->selected_ap_.get_bssid();
515  float priority = this->get_sta_priority(bssid);
516  this->set_sta_priority(bssid, priority - 1.0f);
517  }
518 
519  delay(10);
520  if (!this->is_captive_portal_active_() && (this->num_retried_ > 5 || this->error_from_callback_)) {
521  // If retry failed for more than 5 times, let's restart STA
522  ESP_LOGW(TAG, "Restarting WiFi adapter...");
523  this->wifi_mode_(false, {});
524  delay(100); // NOLINT
525  this->num_retried_ = 0;
526  } else {
527  this->num_retried_++;
528  }
529  this->error_from_callback_ = false;
530  if (this->state_ == WIFI_COMPONENT_STATE_STA_CONNECTING) {
531  yield();
532  this->state_ = WIFI_COMPONENT_STATE_STA_CONNECTING_2;
533  this->start_connecting(this->selected_ap_, true);
534  return;
535  }
536 
537  this->state_ = WIFI_COMPONENT_STATE_COOLDOWN;
538  this->action_started_ = millis();
539 }

590 bool WiFiScanResult::matches(const WiFiAP &config) {
591  if (config.get_hidden()) {
592  // User configured a hidden network, only match actually hidden networks
593  // don't match SSID
594  if (!this->is_hidden_)
595  return false;
596  } else if (!config.get_ssid().empty()) {
597  // check if SSID matches
598  if (config.get_ssid() != this->ssid_)
599  return false;
600  } else {
601  // network is configured without SSID - match other settings
602  }
603  // If BSSID configured, only match for correct BSSIDs
604  if (config.get_bssid().has_value() && *config.get_bssid() != this->bssid_)
605  return false;
606 
607 #ifdef ESPHOME_WIFI_WPA2_EAP
608  // BSSID requires auth but no PSK or EAP credentials given
609  if (this->with_auth_ && (config.get_password().empty() && !config.get_eap().has_value()))
610  return false;
611 
612  // BSSID does not require auth, but PSK or EAP credentials given
613  if (!this->with_auth_ && (!config.get_password().empty() || config.get_eap().has_value()))
614  return false;
615 #else
616  // If PSK given, only match for networks with auth (and vice versa)
617  if (config.get_password().empty() == this->with_auth_)
618  return false;
619 #endif
620 
621  // If channel configured, only match networks on that channel.
622  if (config.get_channel().has_value() && *config.get_channel() != this->channel_) {
623  return false;
624  }
625  return true;
626 }
