177  this->status_set_warning();

184  meas_time += 2.3f * oversampling_to_time(this->humidity_oversampling_) + 0.575f;

207 float BME280Component::read_temperature_(int32_t *t_fine) {
208  uint8_t data[3];
209  if (!this->read_bytes(BME280_REGISTER_TEMPDATA, data, 3))
210  return NAN;
211  int32_t adc = ((data[0] & 0xFF) << 16) | ((data[1] & 0xFF) << 8) | (data[2] & 0xFF);
212  adc >>= 4;
213  if (adc == 0x80000)
214  // temperature was disabled
215  return NAN;
216 
217  const int32_t t1 = this->calibration_.t1;
218  const int32_t t2 = this->calibration_.t2;
219  const int32_t t3 = this->calibration_.t3;
220 
221  int32_t var1 = (((adc >> 3) - (t1 << 1)) * t2) >> 11;
222  int32_t var2 = (((((adc >> 4) - t1) * ((adc >> 4) - t1)) >> 12) * t3) >> 14;

234  adc >>= 4;
235  if (adc == 0x80000)
236  // pressure was disabled
237  return NAN;
238  const int64_t p1 = this->calibration_.p1;
239  const int64_t p2 = this->calibration_.p2;
240  const int64_t p3 = this->calibration_.p3;
241  const int64_t p4 = this->calibration_.p4;
242  const int64_t p5 = this->calibration_.p5;
243  const int64_t p6 = this->calibration_.p6;
244  const int64_t p7 = this->calibration_.p7;
245  const int64_t p8 = this->calibration_.p8;
246  const int64_t p9 = this->calibration_.p9;
247 
248  int64_t var1, var2, p;
249  var1 = int64_t(t_fine) - 128000;
250  var2 = var1 * var1 * p6;
251  var2 = var2 + ((var1 * p5) << 17);
252  var2 = var2 + (p4 << 35);
253  var1 = ((var1 * var1 * p3) >> 8) + ((var1 * p2) << 12);
254  var1 = ((int64_t(1) << 47) + var1) * p1 >> 33;
255 
256  if (var1 == 0)
257  return NAN;
258 
259  p = 1048576 - adc;
260  p = (((p << 31) - var2) * 3125) / var1;
261  var1 = (p9 * (p >> 13) * (p >> 13)) >> 25;
262  var2 = (p8 * p) >> 19;
263 
264  p = ((p + var1 + var2) >> 8) + (p7 << 4);
