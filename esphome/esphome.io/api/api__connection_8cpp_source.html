45 void APIConnection::parse_recv_buffer_() {
46  if (this->recv_buffer_.empty() || this->remove_)
47  return;
48 
49  while (!this->recv_buffer_.empty()) {
50  if (this->recv_buffer_[0] != 0x00) {
51  ESP_LOGW(TAG, "Invalid preamble from %s", this->client_info_.c_str());
52  this->on_fatal_error();
53  return;
54  }
55  uint32_t i = 1;
56  const uint32_t size = this->recv_buffer_.size();
57  uint32_t consumed;
58  auto msg_size_varint = ProtoVarInt::parse(&this->recv_buffer_[i], size - i, &consumed);
59  if (!msg_size_varint.has_value())
60  // not enough data there yet
61  return;
62  i += consumed;
63  uint32_t msg_size = msg_size_varint->as_uint32();
64 
65  auto msg_type_varint = ProtoVarInt::parse(&this->recv_buffer_[i], size - i, &consumed);
66  if (!msg_type_varint.has_value())
67  // not enough data there yet
68  return;
69  i += consumed;
70  uint32_t msg_type = msg_type_varint->as_uint32();
71 
72  if (size - i < msg_size)
73  // message body not fully received
74  return;
75 
76  uint8_t *msg = &this->recv_buffer_[i];
77  this->read_message(msg_size, msg_type, msg);
78  if (this->remove_)
79  return;
80  // pop front
81  uint32_t total = i + msg_size;
82  this->recv_buffer_.erase(this->recv_buffer_.begin(), this->recv_buffer_.begin() + total);
83  this->last_traffic_ = millis();
84  }
