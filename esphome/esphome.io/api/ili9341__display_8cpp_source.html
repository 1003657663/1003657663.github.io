86 void ILI9341Display::display_() {
87  // we will only update the changed window to the display
88  int w = this->x_high_ - this->x_low_ + 1;
89  int h = this->y_high_ - this->y_low_ + 1;

94  for (uint16_t row = 0; row < h; row++) {
95  for (uint16_t col = 0; col < w; col++) {
96  uint32_t pos = start_pos + (row * width_) + col;
97 
98  uint16_t color = convert_to_16bit_color_(buffer_[pos]);
99  this->write_byte(color >> 8);

101  }
102  }
103  this->end_data_();
104

107  this->y_low_ = this->height_;

112 uint16_t ILI9341Display::convert_to_16bit_color_(uint8_t color_8bit) {
113  int r = color_8bit >> 5;
114  int g = (color_8bit >> 2) & 0x07;
115  int b = color_8bit & 0x03;
116  uint16_t color = (r * 0x04) << 11;
117  color |= (g * 0x09) << 5;
118  color |= (b * 0x0A);
119 
120  return color;
121 }
