403 static bool topic_match(const char *message, const char *subscription, bool is_normal, bool past_separator) {
404  // Reached end of both strings at the same time, this means we have a successful match
405  if (*message == '\0' && *subscription == '\0')
406  return true;
407 
408  // Either the message or the subscribe are at the end. This means they don't match.
409  if (*message == '\0' || *subscription == '\0')
410  return false;
411 
412  bool do_wildcards = is_normal || past_separator;
413 
414  if (*subscription == '+' && do_wildcards) {
415  // single level wildcard
416  // consume + from subscription
417  subscription++;
418  // consume everything from message until '/' found or end of string
419  while (*message != '\0' && *message != '/') {
420  message++;
421  }
422  // after this, both pointers will point to a '/' or to the end of the string
423 
424  return topic_match(message, subscription, is_normal, true);
425  }
426 
427  if (*subscription == '#' && do_wildcards) {
428  // multilevel wildcard - MQTT mandates that this must be at end of subscribe topic
429  return true;
430  }
431 
432  // this handles '/' and normal characters at the same time.
433  if (*message != *subscription)
434  return false;
435 
436  past_separator = past_separator || *subscription == '/';
437 
438  // consume characters
439  subscription++;
440  message++;
441 
442  return topic_match(message, subscription, is_normal, past_separator);
443 }
