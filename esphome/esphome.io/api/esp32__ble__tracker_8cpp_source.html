371 void ESPBTDevice::parse_adv_(const esp_ble_gap_cb_param_t::ble_scan_result_evt_param &param) {
372  size_t offset = 0;
373  const uint8_t *payload = param.ble_adv;
374  uint8_t len = param.adv_data_len + param.scan_rsp_len;
375 
376  while (offset + 2 < len) {
377  const uint8_t field_length = payload[offset++]; // First byte is length of adv record
378  if (field_length == 0)
379  break;
380 
381  // first byte of adv record is adv record type
382  const uint8_t record_type = payload[offset++];
383  const uint8_t *record = &payload[offset];
384  const uint8_t record_length = field_length - 1;
385  offset += record_length;
386 
387  // See also Generic Access Profile Assigned Numbers:
388  // https://www.bluetooth.com/specifications/assigned-numbers/generic-access-profile/ See also ADVERTISING AND SCAN
389  // RESPONSE DATA FORMAT: https://www.bluetooth.com/specifications/bluetooth-core-specification/ (vol 3, part C, 11)
390  // See also Core Specification Supplement: https://www.bluetooth.com/specifications/bluetooth-core-specification/
391  // (called CSS here)
392 
393  switch (record_type) {
394  case ESP_BLE_AD_TYPE_NAME_CMPL: {
395  // CSS 1.2 LOCAL NAME
396  // "The Local Name data type shall be the same as, or a shortened version of, the local name assigned to the
397  // device." CSS 1: Optional in this context; shall not appear more than once in a block.
398  this->name_ = std::string(reinterpret_cast<const char *>(record), record_length);
399  break;
400  }
401  case ESP_BLE_AD_TYPE_TX_PWR: {
402  // CSS 1.5 TX POWER LEVEL
403  // "The TX Power Level data type indicates the transmitted power level of the packet containing the data type."
404  // CSS 1: Optional in this context (may appear more than once in a block).
405  this->tx_powers_.push_back(*payload);
406  break;
407  }
408  case ESP_BLE_AD_TYPE_APPEARANCE: {
409  // CSS 1.12 APPEARANCE
410  // "The Appearance data type defines the external appearance of the device."
411  // See also https://www.bluetooth.com/specifications/gatt/characteristics/
412  // CSS 1: Optional in this context; shall not appear more than once in a block and shall not appear in both
413  // the AD and SRD of the same extended advertising interval.
414  this->appearance_ = *reinterpret_cast<const uint16_t *>(record);
415  break;
416  }
417  case ESP_BLE_AD_TYPE_FLAG: {
418  // CSS 1.3 FLAGS
419  // "The Flags data type contains one bit Boolean flags. The Flags data type shall be included when any of the
420  // Flag bits are non-zero and the advertising packet is connectable, otherwise the Flags data type may be
421  // omitted."
422  // CSS 1: Optional in this context; shall not appear more than once in a block.
423  this->ad_flag_ = *record;
424  break;
425  }
426  // CSS 1.1 SERVICE UUID
427  // The Service UUID data type is used to include a list of Service or Service Class UUIDs.
428  // There are six data types defined for the three sizes of Service UUIDs that may be returned:
429  // CSS 1: Optional in this context (may appear more than once in a block).
430  case ESP_BLE_AD_TYPE_16SRV_CMPL:
431  case ESP_BLE_AD_TYPE_16SRV_PART: {
432  // • 16-bit Bluetooth Service UUIDs
433  for (uint8_t i = 0; i < record_length / 2; i++) {
434  this->service_uuids_.push_back(ESPBTUUID::from_uint16(*reinterpret_cast<const uint16_t *>(record + 2 * i)));
435  }
436  break;
437  }
438  case ESP_BLE_AD_TYPE_32SRV_CMPL:
439  case ESP_BLE_AD_TYPE_32SRV_PART: {
440  // • 32-bit Bluetooth Service UUIDs
441  for (uint8_t i = 0; i < record_length / 4; i++) {
442  this->service_uuids_.push_back(ESPBTUUID::from_uint32(*reinterpret_cast<const uint32_t *>(record + 4 * i)));
443  }
444  break;
445  }
446  case ESP_BLE_AD_TYPE_128SRV_CMPL:
447  case ESP_BLE_AD_TYPE_128SRV_PART: {
448  // • Global 128-bit Service UUIDs
449  this->service_uuids_.push_back(ESPBTUUID::from_raw(record));
450  break;
451  }
452  case ESP_BLE_AD_MANUFACTURER_SPECIFIC_TYPE: {
453  // CSS 1.4 MANUFACTURER SPECIFIC DATA
454  // "The Manufacturer Specific data type is used for manufacturer specific data. The first two data octets shall
455  // contain a company identifier from Assigned Numbers. The interpretation of any other octets within the data
456  // shall be defined by the manufacturer specified by the company identifier."
457  // CSS 1: Optional in this context (may appear more than once in a block).
458  if (record_length < 2) {
459  ESP_LOGV(TAG, "Record length too small for ESP_BLE_AD_MANUFACTURER_SPECIFIC_TYPE");
460  break;
461  }
462  ServiceData data{};
463  data.uuid = ESPBTUUID::from_uint16(*reinterpret_cast<const uint16_t *>(record));
464  data.data.assign(record + 2UL, record + record_length);
465  this->manufacturer_datas_.push_back(data);
466  break;
467  }
468 
469  // CSS 1.11 SERVICE DATA
470  // "The Service Data data type consists of a service UUID with the data associated with that service."
471  // CSS 1: Optional in this context (may appear more than once in a block).
472  case ESP_BLE_AD_TYPE_SERVICE_DATA: {
473  // «Service Data - 16 bit UUID»
474  // Size: 2 or more octets
475  // The first 2 octets contain the 16 bit Service UUID fol- lowed by additional service data
476  if (record_length < 2) {
477  ESP_LOGV(TAG, "Record length too small for ESP_BLE_AD_TYPE_SERVICE_DATA");
478  break;
479  }
480  ServiceData data{};
481  data.uuid = ESPBTUUID::from_uint16(*reinterpret_cast<const uint16_t *>(record));
482  data.data.assign(record + 2UL, record + record_length);
483  this->service_datas_.push_back(data);
484  break;
485  }
486  case ESP_BLE_AD_TYPE_32SERVICE_DATA: {
487  // «Service Data - 32 bit UUID»
488  // Size: 4 or more octets
489  // The first 4 octets contain the 32 bit Service UUID fol- lowed by additional service data
490  if (record_length < 4) {
491  ESP_LOGV(TAG, "Record length too small for ESP_BLE_AD_TYPE_32SERVICE_DATA");
492  break;
493  }
494  ServiceData data{};
495  data.uuid = ESPBTUUID::from_uint32(*reinterpret_cast<const uint32_t *>(record));
496  data.data.assign(record + 4UL, record + record_length);
497  this->service_datas_.push_back(data);
498  break;
499  }
500  case ESP_BLE_AD_TYPE_128SERVICE_DATA: {
501  // «Service Data - 128 bit UUID»
502  // Size: 16 or more octets
503  // The first 16 octets contain the 128 bit Service UUID followed by additional service data
504  if (record_length < 16) {
505  ESP_LOGV(TAG, "Record length too small for ESP_BLE_AD_TYPE_128SERVICE_DATA");
506  break;
507  }
508  ServiceData data{};
509  data.uuid = ESPBTUUID::from_raw(record);
510  data.data.assign(record + 16UL, record + record_length);
511  this->service_datas_.push_back(data);
512  break;
513  }
514  default: {
515  ESP_LOGV(TAG, "Unhandled type: advType: 0x%02x", record_type);
516  break;
517  }
518  }
519  }
