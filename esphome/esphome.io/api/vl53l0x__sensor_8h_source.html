216  uint16_t decode_timeout_(uint16_t reg_val) {
217  // format: "(LSByte * 2^MSByte) + 1"
218  uint8_t msb = (reg_val >> 8) & 0xFF;
219  uint8_t lsb = (reg_val >> 0) & 0xFF;
220  return (uint16_t(lsb) << msb) + 1;

222  uint16_t encode_timeout_(uint16_t timeout_mclks) {
223  // format: "(LSByte * 2^MSByte) + 1"
224  uint32_t ls_byte = 0;
225  uint16_t ms_byte = 0;
226 
227  if (timeout_mclks <= 0)
228  return 0;
229 
230  ls_byte = timeout_mclks - 1;
231 
232  while ((ls_byte & 0xFFFFFF00) > 0) {
233  ls_byte >>= 1;
234  ms_byte++;
235  }
