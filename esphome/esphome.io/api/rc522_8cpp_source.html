66 void RC522::initialize_() {
67  // Per original code, wait 50 ms
68  if (millis() - reset_timeout_ < 50)
69  return;
70 
71  // Reset baud rates
72  ESP_LOGV(TAG, "Initialize");
73 
74  pcd_write_register(TX_MODE_REG, 0x00);
75  pcd_write_register(RX_MODE_REG, 0x00);
76  // Reset ModWidthReg
77  pcd_write_register(MOD_WIDTH_REG, 0x26);
78 
79  // When communicating with a PICC we need a timeout if something goes wrong.
80  // f_timer = 13.56 MHz / (2*TPreScaler+1) where TPreScaler = [TPrescaler_Hi:TPrescaler_Lo].
81  // TPrescaler_Hi are the four low bits in TModeReg. TPrescaler_Lo is TPrescalerReg.
82  pcd_write_register(T_MODE_REG, 0x80); // TAuto=1; timer starts automatically at the end of the transmission in all
83  // communication modes at all speeds
84 
85  // TPreScaler = TModeReg[3..0]:TPrescalerReg, ie 0x0A9 = 169 => f_timer=40kHz, ie a timer period of 25μs.
86  pcd_write_register(T_PRESCALER_REG, 0xA9);
87  pcd_write_register(T_RELOAD_REG_H, 0x03); // Reload timer with 0x3E8 = 1000, ie 25ms before timeout.
88  pcd_write_register(T_RELOAD_REG_L, 0xE8);
89 
90  // Default 0x00. Force a 100 % ASK modulation independent of the ModGsPReg register setting
91  pcd_write_register(TX_ASK_REG, 0x40);
92  pcd_write_register(MODE_REG, 0x3D); // Default 0x3F. Set the preset value for the CRC coprocessor for the CalcCRC

278 void RC522::pcd_reset_() {
279  // The datasheet does not mention how long the SoftRest command takes to complete.
280  // But the MFRC522 might have been in soft power-down mode (triggered by bit 4 of CommandReg)
281  // Section 8.8.2 in the datasheet says the oscillator start-up time is the start up time of the crystal + 37,74μs.
282  // Let us be generous: 50ms.
283 
284  if (millis() - reset_timeout_ < 50)
285  return;
286 
287  if (reset_count_ == RESET_COUNT) {
288  ESP_LOGI(TAG, "Soft reset...");
289  // Issue the SoftReset command.
290  pcd_write_register(COMMAND_REG, PCD_SOFT_RESET);
291  }
292 
293  // Expect the PowerDown bit in CommandReg to be cleared (max 3x50ms)
294  if ((pcd_read_register(COMMAND_REG) & (1 << 4)) == 0) {
295  reset_count_ = 0;
296  ESP_LOGI(TAG, "Device online.");
297  // Wait for initialize
298  reset_timeout_ = millis();
299  return;
300  }
301 
302  if (--reset_count_ == 0) {
303  ESP_LOGE(TAG, "Unable to reset RC522.");
304  this->error_code_ = RESET_FAILED;
305  mark_failed();
306  }
307 }

356 void RC522::pcd_transceive_data_(uint8_t send_len) {
357  ESP_LOGV(TAG, "PCD TRANSCEIVE: RX: %s", format_buffer(buffer_, send_len).c_str());
358  delayMicroseconds(1000); // we need 1 ms delay between antenna on and those communication commands
359  send_len_ = send_len;
360  // Prepare values for BitFramingReg
361  // For REQA and WUPA we need the short frame format - transmit only 7 bits of the last (and only)
362  // uint8_t. TxLastBits = BitFramingReg[2..0]
363  uint8_t bit_framing = (buffer_[0] == PICC_CMD_REQA) ? 7 : 0;
364 
365  pcd_write_register(COMMAND_REG, PCD_IDLE); // Stop any active command.
366  pcd_write_register(COM_IRQ_REG, 0x7F); // Clear all seven interrupt request bits
367  pcd_write_register(FIFO_LEVEL_REG, 0x80); // FlushBuffer = 1, FIFO initialization
368  pcd_write_register(FIFO_DATA_REG, send_len_, buffer_); // Write sendData to the FIFO
369  pcd_write_register(BIT_FRAMING_REG, bit_framing); // Bit adjustments
370  pcd_write_register(COMMAND_REG, PCD_TRANSCEIVE); // Execute the command
371  pcd_set_register_bit_mask_(BIT_FRAMING_REG, 0x80); // StartSend=1, transmission of data starts
372  awaiting_comm_ = true;
373  awaiting_comm_time_ = millis();
374 }

376 RC522::StatusCode RC522::await_transceive_() {
377  if (millis() - awaiting_comm_time_ < 2) // wait at least 2 ms
378  return STATUS_WAITING;
379  uint8_t n = pcd_read_register(
380  COM_IRQ_REG); // ComIrqReg[7..0] bits are: Set1 TxIRq RxIRq IdleIRq HiAlertIRq LoAlertIRq ErrIRq TimerIRq
381  if (n & 0x01) { // Timer interrupt - nothing received in 25ms
382  back_length_ = 0;
383  error_counter_ = 0; // reset the error counter
384  return STATUS_TIMEOUT;
385  }
386  if (!(n & WAIT_I_RQ)) { // None of the interrupts that signal success has been set.
387  // Wait for the command to complete.
388  if (millis() - awaiting_comm_time_ < 40)
389  return STATUS_WAITING;
390  back_length_ = 0;
391  ESP_LOGW(TAG, "Communication with the MFRC522 might be down, reset in %d",
392  10 - error_counter_); // todo: trigger reset?
393  if (error_counter_++ > 10)
394  setup();
395 
396  return STATUS_TIMEOUT;
397  }
398  // Stop now if any errors except collisions were detected.
399  uint8_t error_reg_value = pcd_read_register(
400  ERROR_REG); // ErrorReg[7..0] bits are: WrErr TempErr reserved BufferOvfl CollErr CRCErr ParityErr ProtocolErr
401  if (error_reg_value & 0x13) { // BufferOvfl ParityErr ProtocolErr
402  return STATUS_ERROR;
403  }
404  error_counter_ = 0; // reset the error counter
405 
406  n = pcd_read_register(FIFO_LEVEL_REG); // Number of uint8_ts in the FIFO
407  if (n > sizeof(buffer_))
408  return STATUS_NO_ROOM;
409  if (n > sizeof(buffer_) - send_len_)
410  send_len_ = sizeof(buffer_) - n; // simply overwrite the sent values
411  back_length_ = n; // Number of uint8_ts returned
412  pcd_read_register(FIFO_DATA_REG, n, buffer_ + send_len_, rx_align_); // Get received data from FIFO
413  uint8_t valid_bits_local =
414  pcd_read_register(CONTROL_REG) & 0x07; // RxLastBits[2:0] indicates the number of valid bits in the last
415  // received uint8_t. If this value is 000b, the whole uint8_t is valid.
416 
417  // Tell about collisions
418  if (error_reg_value & 0x08) { // CollErr
419  ESP_LOGW(TAG, "collision error, received %d bytes + %d bits (but anticollision not implemented)",
420  back_length_ - (valid_bits_local > 0), valid_bits_local);
421  return STATUS_COLLISION;
422  }
423  // Tell about collisions
424  if (valid_bits_local) {
425  ESP_LOGW(TAG, "only %d valid bits received, tag distance to high? Error code is 0x%x", valid_bits_local,
426  error_reg_value); // TODO: is this always due to collissions?
427  return STATUS_ERROR;
428  }
429  ESP_LOGV(TAG, "received %d bytes: %s", back_length_, format_buffer(buffer_ + send_len_, back_length_).c_str());
430 
431  return STATUS_OK;
432 }

455  if (millis() - awaiting_comm_time_ < 2) // wait at least 2 ms
456  return STATUS_WAITING;
