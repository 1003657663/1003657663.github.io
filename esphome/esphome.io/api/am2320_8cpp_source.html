15 uint16_t crc_16(uint8_t *ptr, uint8_t length) {
16  uint16_t crc = 0xFFFF;
17  uint8_t i;
18  //------------------------------
19  while (length--) {
20  crc ^= *ptr++;
21  for (i = 0; i < 8; i++)
22  if ((crc & 0x01) != 0) {
23  crc >>= 1;
24  crc ^= 0xA001;
25  } else
26  crc >>= 1;
27  }
28  return crc;
29 }
