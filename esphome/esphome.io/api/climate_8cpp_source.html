38 void ClimateCall::validate_() {
39  auto traits = this->parent_->get_traits();
40  if (this->mode_.has_value()) {
41  auto mode = *this->mode_;
42  if (!traits.supports_mode(mode)) {
43  ESP_LOGW(TAG, " Mode %s is not supported by this device!", climate_mode_to_string(mode));
44  this->mode_.reset();
45  }
46  }
47  if (this->fan_mode_.has_value()) {
48  auto fan_mode = *this->fan_mode_;
49  if (!traits.supports_fan_mode(fan_mode)) {
50  ESP_LOGW(TAG, " Fan Mode %s is not supported by this device!", climate_fan_mode_to_string(fan_mode));
51  this->fan_mode_.reset();
52  }
53  }
54  if (this->swing_mode_.has_value()) {
55  auto swing_mode = *this->swing_mode_;
56  if (!traits.supports_swing_mode(swing_mode)) {
57  ESP_LOGW(TAG, " Swing Mode %s is not supported by this device!", climate_swing_mode_to_string(swing_mode));
58  this->swing_mode_.reset();
59  }
60  }
61  if (this->target_temperature_.has_value()) {
62  auto target = *this->target_temperature_;
63  if (traits.get_supports_two_point_target_temperature()) {
64  ESP_LOGW(TAG, " Cannot set target temperature for climate device "
65  "with two-point target temperature!");
66  this->target_temperature_.reset();
67  } else if (isnan(target)) {
68  ESP_LOGW(TAG, " Target temperature must not be NAN!");
69  this->target_temperature_.reset();
70  }
71  }
72  if (this->target_temperature_low_.has_value() || this->target_temperature_high_.has_value()) {
73  if (!traits.get_supports_two_point_target_temperature()) {
74  ESP_LOGW(TAG, " Cannot set low/high target temperature for this device!");
75  this->target_temperature_low_.reset();
76  this->target_temperature_high_.reset();
77  }
78  }
79  if (this->target_temperature_low_.has_value() && isnan(*this->target_temperature_low_)) {
80  ESP_LOGW(TAG, " Target temperature low must not be NAN!");
81  this->target_temperature_low_.reset();
82  }
83  if (this->target_temperature_high_.has_value() && isnan(*this->target_temperature_high_)) {
84  ESP_LOGW(TAG, " Target temperature low must not be NAN!");
85  this->target_temperature_high_.reset();
86  }
87  if (this->target_temperature_low_.has_value() && this->target_temperature_high_.has_value()) {
88  float low = *this->target_temperature_low_;
89  float high = *this->target_temperature_high_;
90  if (low > high) {
91  ESP_LOGW(TAG, " Target temperature low %.2f must be smaller than target temperature high %.2f!", low, high);
92  this->target_temperature_low_.reset();
93  this->target_temperature_high_.reset();
94  }
95  }
96  if (this->away_.has_value()) {
97  if (!traits.get_supports_away()) {
98  ESP_LOGW(TAG, " Cannot set away mode for this device!");
99  this->away_.reset();
100  }
101  }
