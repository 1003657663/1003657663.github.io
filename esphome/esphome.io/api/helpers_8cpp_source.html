159 uint8_t crc8(uint8_t *data, uint8_t len) {
160  uint8_t crc = 0;
161 
162  while ((len--) != 0u) {
163  uint8_t inbyte = *data++;
164  for (uint8_t i = 8; i != 0u; i--) {
165  bool mix = (crc ^ inbyte) & 0x01;
166  crc >>= 1;
167  if (mix)
168  crc ^= 0x8C;
169  inbyte >>= 1;
170  }
171  }
172  return crc;
173 }
