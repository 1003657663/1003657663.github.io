205 void FujitsuGeneralClimate::transmit_(uint8_t const* message, uint8_t length) {
206  ESP_LOGV(TAG, "Transmit message length %d", length);
207 
208  auto transmit = this->transmitter_->transmit();
209  auto data = transmit.get_data();
210 
211  data->set_carrier_frequency(FUJITSU_GENERAL_CARRIER_FREQUENCY);
212 
213  // Header
214  data->mark(FUJITSU_GENERAL_HEADER_MARK);
215  data->space(FUJITSU_GENERAL_HEADER_SPACE);
216 
217  // Data
218  for (uint8_t i = 0; i < length; ++i) {
219  const uint8_t byte = message[i];
220  for (uint8_t mask = 0b00000001; mask > 0; mask <<= 1) { // write from right to left
221  data->mark(FUJITSU_GENERAL_BIT_MARK);
222  bool bit = byte & mask;
223  data->space(bit ? FUJITSU_GENERAL_ONE_SPACE : FUJITSU_GENERAL_ZERO_SPACE);
224  }
225  }
226 
227  // Footer
228  data->mark(FUJITSU_GENERAL_TRL_MARK);
229  data->space(FUJITSU_GENERAL_TRL_SPACE);
230 
231  transmit.perform();
232 }

244 bool FujitsuGeneralClimate::on_receive(remote_base::RemoteReceiveData data) {
245  ESP_LOGV(TAG, "Received IR message");
246 
247  // Validate header
248  if (!data.expect_item(FUJITSU_GENERAL_HEADER_MARK, FUJITSU_GENERAL_HEADER_SPACE)) {
249  ESP_LOGV(TAG, "Header fail");
250  return false;
251  }
252 
253  uint8_t recv_message[FUJITSU_GENERAL_STATE_MESSAGE_LENGTH] = {0};
254 
255  // Read header
256  for (uint8_t byte = 0; byte < FUJITSU_GENERAL_COMMON_LENGTH; ++byte) {
257  // Read bit
258  for (uint8_t bit = 0; bit < 8; ++bit) {
259  if (data.expect_item(FUJITSU_GENERAL_BIT_MARK, FUJITSU_GENERAL_ONE_SPACE)) {
260  recv_message[byte] |= 1 << bit; // read from right to left
261  } else if (!data.expect_item(FUJITSU_GENERAL_BIT_MARK, FUJITSU_GENERAL_ZERO_SPACE)) {
262  ESP_LOGV(TAG, "Byte %d bit %d fail", byte, bit);
263  return false;
264  }
265  }
266  }
267 
268  const uint8_t recv_message_type = recv_message[FUJITSU_GENERAL_MESSAGE_TYPE_BYTE];
269  uint8_t recv_message_length;
270 
271  switch (recv_message_type) {
272  case FUJITSU_GENERAL_MESSAGE_TYPE_STATE:
273  ESP_LOGV(TAG, "Received state message");
274  recv_message_length = FUJITSU_GENERAL_STATE_MESSAGE_LENGTH;
275  break;
276  case FUJITSU_GENERAL_MESSAGE_TYPE_OFF:
277  case FUJITSU_GENERAL_MESSAGE_TYPE_ECONOMY:
278  case FUJITSU_GENERAL_MESSAGE_TYPE_NUDGE:
279  ESP_LOGV(TAG, "Received util message");
280  recv_message_length = FUJITSU_GENERAL_UTIL_MESSAGE_LENGTH;
281  break;
282  default:
283  ESP_LOGV(TAG, "Unknown message type %X", recv_message_type);
284  return false;
285  }
286 
287  // Read message body
288  for (uint8_t byte = FUJITSU_GENERAL_COMMON_LENGTH; byte < recv_message_length; ++byte) {
289  for (uint8_t bit = 0; bit < 8; ++bit) {
290  if (data.expect_item(FUJITSU_GENERAL_BIT_MARK, FUJITSU_GENERAL_ONE_SPACE)) {
291  recv_message[byte] |= 1 << bit; // read from right to left
292  } else if (!data.expect_item(FUJITSU_GENERAL_BIT_MARK, FUJITSU_GENERAL_ZERO_SPACE)) {
293  ESP_LOGV(TAG, "Byte %d bit %d fail", byte, bit);
294  return false;
295  }
296  }
297  }
298 
299  // Validate footer
300  if (!data.expect_mark(FUJITSU_GENERAL_BIT_MARK)) {
301  ESP_LOGV(TAG, "Footer fail");
302  return false;
303  }
304 
305  for (uint8_t byte = 0; byte < recv_message_length; ++byte) {
306  ESP_LOGVV(TAG, "%02X", recv_message[byte]);
307  }
308 
309  const uint8_t recv_checksum = recv_message[recv_message_length - 1];
310  uint8_t calculated_checksum;
311  if (recv_message_type == FUJITSU_GENERAL_MESSAGE_TYPE_STATE) {
312  calculated_checksum = this->checksum_state_(recv_message);
313  } else {
314  calculated_checksum = this->checksum_util_(recv_message);
315  }
316 
317  if (recv_checksum != calculated_checksum) {
318  ESP_LOGV(TAG, "Checksum fail - expected %X - got %X", calculated_checksum, recv_checksum);
319  return false;
320  }
321 
322  if (recv_message_type == FUJITSU_GENERAL_MESSAGE_TYPE_STATE) {
323  const uint8_t recv_tempertature = GET_NIBBLE(recv_message, FUJITSU_GENERAL_TEMPERATURE_NIBBLE);
324  const uint8_t offset_temperature = recv_tempertature + FUJITSU_GENERAL_TEMP_MIN;
325  this->target_temperature = offset_temperature;
326  ESP_LOGV(TAG, "Received temperature %d", offset_temperature);
327 
328  const uint8_t recv_mode = GET_NIBBLE(recv_message, FUJITSU_GENERAL_MODE_NIBBLE);
329  ESP_LOGV(TAG, "Received mode %X", recv_mode);
330  switch (recv_mode) {
331  case FUJITSU_GENERAL_MODE_COOL:
332  this->mode = climate::CLIMATE_MODE_COOL;
333  break;
334  case FUJITSU_GENERAL_MODE_HEAT:
335  this->mode = climate::CLIMATE_MODE_HEAT;
336  break;
337  case FUJITSU_GENERAL_MODE_DRY:
338  this->mode = climate::CLIMATE_MODE_DRY;
339  break;
340  case FUJITSU_GENERAL_MODE_FAN:
341  this->mode = climate::CLIMATE_MODE_FAN_ONLY;
342  break;
343  case FUJITSU_GENERAL_MODE_AUTO:
344  default:
345  // TODO: CLIMATE_MODE_10C is missing from esphome
346  this->mode = climate::CLIMATE_MODE_AUTO;
347  break;
348  }
349 
350  const uint8_t recv_fan_mode = GET_NIBBLE(recv_message, FUJITSU_GENERAL_FAN_NIBBLE);
351  ESP_LOGV(TAG, "Received fan mode %X", recv_fan_mode);
352  switch (recv_fan_mode) {
