239  bool full_update = this->at_update_ == 0;
240  bool prev_full_update = this->at_update_ == 1;
241 
242  if (!this->wait_until_idle_()) {
243  this->status_set_warning();
244  return;
245  }
246 
247  if (this->full_update_every_ >= 1) {
248  if (full_update != prev_full_update) {
249  switch (this->model_) {
250  case TTGO_EPAPER_2_13_IN:
251  this->write_lut_(full_update ? FULL_UPDATE_LUT_TTGO : PARTIAL_UPDATE_LUT_TTGO, LUT_SIZE_TTGO);
252  break;
253  case TTGO_EPAPER_2_13_IN_B73:
254  this->write_lut_(full_update ? FULL_UPDATE_LUT_TTGO_B73 : PARTIAL_UPDATE_LUT_TTGO_B73, LUT_SIZE_TTGO_B73);
255  break;
256  case TTGO_EPAPER_2_13_IN_B1:
257  this->write_lut_(full_update ? FULL_UPDATE_LUT_TTGO_B1 : PARTIAL_UPDATE_LUT_TTGO_B1, LUT_SIZE_TTGO_B1);
258  break;
259  default:
260  this->write_lut_(full_update ? FULL_UPDATE_LUT : PARTIAL_UPDATE_LUT, LUT_SIZE_WAVESHARE);
261  }
262  }
263  this->at_update_ = (this->at_update_ + 1) % this->full_update_every_;
264  }
265

267  switch (this->model_) {
268  case TTGO_EPAPER_2_13_IN_B1:
269  // COMMAND SET RAM X ADDRESS START END POSITION
270  this->command(0x44);

274  this->command(0x45);

277  this->data(0x00);

287 
288  break;
289

293  this->data(0x00);
