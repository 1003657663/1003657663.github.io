193  bool r;
194  {
195  InterruptLock lock;
196  r = this->read_scratch_pad();
197  }
198 
199  if (!r) {
200  ESP_LOGE(TAG, "Reading scratchpad failed: reset");
201  return false;
202  }

204  return false;
205

207  return false;
208

210  // DS18S20 doesn't support resolution.
211  ESP_LOGW(TAG, "DS18S20 doesn't support setting resolution.");
212  return false;
213  }
214 
215  switch (this->resolution_) {
216  case 12:

218  break;
219  case 11:

222  case 10:
223  this->scratch_pad_[4] = 0x3F;
224  break;
225  case 9:

227  this->scratch_pad_[4] = 0x1F;
228  break;
229  }
230

232  {
233  InterruptLock lock;

236  wire->write8(DALLAS_COMMAND_WRITE_SCRATCH_PAD);
237  wire->write8(this->scratch_pad_[2]); // high alarm temp
238  wire->write8(this->scratch_pad_[3]); // low alarm temp
