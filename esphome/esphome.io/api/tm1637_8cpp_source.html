186 bool TM1637Display::send_byte_(uint8_t b) {
187  uint8_t data = b;
188 
189  // 8 Data Bits
190  for (uint8_t i = 0; i < 8; i++) {
191  // CLK low
192  this->clk_pin_->pin_mode(OUTPUT);
193  this->bit_delay_();
194 
195  // Set data bit
196  if (data & 0x01)
197  this->dio_pin_->pin_mode(INPUT);
198  else
199  this->dio_pin_->pin_mode(OUTPUT);
200 
201  this->bit_delay_();
202 
203  // CLK high
204  this->clk_pin_->pin_mode(INPUT);
205  this->bit_delay_();
206  data = data >> 1;
207  }
208 
209  // Wait for acknowledge
210  // CLK to zero
211  this->clk_pin_->pin_mode(OUTPUT);
212  this->dio_pin_->pin_mode(INPUT);
213  this->bit_delay_();
214 
215  // CLK to high
216  this->clk_pin_->pin_mode(INPUT);
217  this->bit_delay_();
218  uint8_t ack = this->dio_pin_->digital_read();
219  if (ack == 0) {
220  this->dio_pin_->pin_mode(OUTPUT);
221  }
222 
223  this->bit_delay_();
224  this->clk_pin_->pin_mode(OUTPUT);
225  this->bit_delay_();
226 
227  return ack;
228 }

230 uint8_t TM1637Display::print(uint8_t start_pos, const char* str) {
231  ESP_LOGV(TAG, "Print at %d: %s", start_pos, str);
232  uint8_t pos = start_pos;
233  for (; *str != '\0'; str++) {
234  uint8_t data = TM1637_UNKNOWN_CHAR;
235  if (*str >= ' ' && *str <= '~')
236  data = pgm_read_byte(&TM1637_ASCII_TO_RAW[*str - ' ']);
237 
238  if (data == TM1637_UNKNOWN_CHAR) {
239  ESP_LOGW(TAG, "Encountered character '%c' with no TM1637 representation while translating string!", *str);
240  }
241  // Remap segments, for compatibility with MAX7219 segment definition which is
242  // XABCDEFG, but TM1637 is // XGFEDCBA
243  data = ((data & 0x80) ? 0x80 : 0) | // no move X
244  ((data & 0x40) ? 0x1 : 0) | // A
245  ((data & 0x20) ? 0x2 : 0) | // B
246  ((data & 0x10) ? 0x4 : 0) | // C
247  ((data & 0x8) ? 0x8 : 0) | // D
248  ((data & 0x4) ? 0x10 : 0) | // E
249  ((data & 0x2) ? 0x20 : 0) | // F
250  ((data & 0x1) ? 0x40 : 0); // G
251  if (*str == '.') {
252  if (pos != start_pos)
253  pos--;
254  this->buffer_[pos] |= 0b10000000;
255  } else {
256  if (pos >= 4) {
257  ESP_LOGE(TAG, "String is too long for the display!");
258  break;
259  }
260  this->buffer_[pos] = data;
261  }
262  pos++;
263  }
264  return pos - start_pos;
265 }
