37  bool current_ap = current_mode & 0b10;
38  bool target_sta = sta.value_or(current_sta);
39  bool target_ap = ap.value_or(current_ap);
40  if (current_sta == target_sta && current_ap == target_ap)
41  return true;
42 
43  if (target_sta && !current_sta) {
44  ESP_LOGV(TAG, "Enabling STA.");

47  // Stop DHCP client when disabling STA

49  wifi_station_dhcpc_stop();

625 bool WiFiComponent::wifi_ap_ip_config_(optional<ManualIP> manual_ip) {
626  // enable AP
627  if (!this->wifi_mode_({}, true))
628  return false;
629 
630  struct ip_info info {};
631  if (manual_ip.has_value()) {
632  info.ip.addr = static_cast<uint32_t>(manual_ip->static_ip);
633  info.gw.addr = static_cast<uint32_t>(manual_ip->gateway);
634  info.netmask.addr = static_cast<uint32_t>(manual_ip->subnet);
635  } else {
636  info.ip.addr = static_cast<uint32_t>(IPAddress(192, 168, 4, 1));
637  info.gw.addr = static_cast<uint32_t>(IPAddress(192, 168, 4, 1));
638  info.netmask.addr = static_cast<uint32_t>(IPAddress(255, 255, 255, 0));
639  }
640 
641  if (wifi_softap_dhcps_status() == DHCP_STARTED) {
642  if (!wifi_softap_dhcps_stop()) {
643  ESP_LOGV(TAG, "Stopping DHCP server failed!");
644  }
645  }
646 
647  if (!wifi_set_ip_info(SOFTAP_IF, &info)) {
648  ESP_LOGV(TAG, "Setting SoftAP info failed!");
649  return false;
650  }
651 
652  struct dhcps_lease lease {};
653  IPAddress start_address = info.ip.addr;
654  start_address[3] += 99;
655  lease.start_ip.addr = static_cast<uint32_t>(start_address);
656  ESP_LOGV(TAG, "DHCP server IP lease start: %s", start_address.toString().c_str());
657  start_address[3] += 100;
658  lease.end_ip.addr = static_cast<uint32_t>(start_address);
659  ESP_LOGV(TAG, "DHCP server IP lease end: %s", start_address.toString().c_str());
660  if (!wifi_softap_set_dhcps_lease(&lease)) {
661  ESP_LOGV(TAG, "Setting SoftAP DHCP lease failed!");
662  return false;
663  }
664 
665  // lease time 1440 minutes (=24 hours)
666  if (!wifi_softap_set_dhcps_lease_time(1440)) {
667  ESP_LOGV(TAG, "Setting SoftAP DHCP lease time failed!");
668  return false;
669  }
670 
671  uint8_t mode = 1;
672  // bit0, 1 enables router information from ESP8266 SoftAP DHCP server.
673  if (!wifi_softap_set_dhcps_offer_option(OFFER_ROUTER, &mode)) {
674  ESP_LOGV(TAG, "wifi_softap_set_dhcps_offer_option failed!");
675  return false;
676  }
677 
678  if (!wifi_softap_dhcps_start()) {
679  ESP_LOGV(TAG, "Starting SoftAP DHCPS failed!");
680  return false;
681  }
682 
683  return true;
684 }
