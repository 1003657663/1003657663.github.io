60  void swap(optional &rhs) {
61  using std::swap;
62  if (has_value() && rhs.has_value()) {
63  swap(**this, *rhs);
64  } else if (!has_value() && rhs.has_value()) {
65  initialize(*rhs);
66  rhs.reset();
67  } else if (has_value() && !rhs.has_value()) {
68  rhs.initialize(**this);
