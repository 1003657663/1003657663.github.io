243  this->gesture_up_started_ = false;
244  this->gesture_down_started_ = false;
245  this->gesture_left_started_ = false;
246  this->gesture_right_started_ = false;

293 void APDS9960::process_dataset_(int up, int down, int left, int right) {
294  /* Algorithm: (see Figure 11 in datasheet)
295  *
296  * Observation: When a gesture is started, we will see a short amount of time where
297  * the photodiode in the direction of the motion has a much higher count value
298  * than where the gesture originates.
299  *
300  * In this algorithm we continually check the difference between the count values of opposing
301  * directions. For example in the down/up direction we continually look at the difference of the
302  * up count and down count. When DOWN gesture begins, this difference will be positive with a
303  * high magnitude for a short amount of time (magic value here is the difference is at least 13).
304  *
305  * If we see such a pattern, we store that we saw the first part of a gesture (the leading edge).
306  * After that some time can pass during which the difference is zero again (though the count values
307  * are not zero). At the end of a gesture, we will see this difference go into the opposite direction
308  * for a short period of time.
309  *
310  * If a gesture is not ended within 500 milliseconds, we consider the initial trailing edge invalid
311  * and reset the state.
312  *
313  * This algorithm does work, but not too well. Some good signal processing algorithms could
314  * probably improve this a lot, especially since the incoming signal has such a characteristic
315  * and quite noise-free pattern.
316  */
317  const int up_down_delta = up - down;
318  const int left_right_delta = left - right;
319  const bool up_down_significant = abs(up_down_delta) > 13;
320  const bool left_right_significant = abs(left_right_delta) > 13;
321 
322  if (up_down_significant) {
323  if (up_down_delta < 0) {
324  if (this->gesture_up_started_) {
325  // trailing edge of gesture up
326  this->report_gesture_(1); // UP
327  } else {
328  // leading edge of gesture down
329  this->gesture_down_started_ = true;
330  this->gesture_start_ = millis();
331  }
332  } else {
333  if (this->gesture_down_started_) {
334  // trailing edge of gesture down
335  this->report_gesture_(2); // DOWN
336  } else {
337  // leading edge of gesture up
338  this->gesture_up_started_ = true;
339  this->gesture_start_ = millis();
340  }
341  }
342  }
343 
344  if (left_right_significant) {
345  if (left_right_delta < 0) {
346  if (this->gesture_left_started_) {
347  // trailing edge of gesture left
348  this->report_gesture_(3); // LEFT
349  } else {
350  // leading edge of gesture right
351  this->gesture_right_started_ = true;
352  this->gesture_start_ = millis();
353  }
354  } else {
355  if (this->gesture_right_started_) {
356  // trailing edge of gesture right
357  this->report_gesture_(4); // RIGHT
358  } else {
359  // leading edge of gesture left
360  this->gesture_left_started_ = true;
361  this->gesture_start_ = millis();
362  }
363  }
364  }
