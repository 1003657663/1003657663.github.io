212 void ICACHE_RAM_ATTR ESP8266SoftwareSerial::gpio_intr(ESP8266SoftwareSerial *arg) {
213  uint32_t wait = arg->bit_time_ + arg->bit_time_ / 3 - 500;
214  const uint32_t start = ESP.getCycleCount();
215  uint8_t rec = 0;
216  // Manually unroll the loop
217  for (int i = 0; i < arg->data_bits_; i++)
218  rec |= arg->read_bit_(&wait, start) << i;
219 
220  /* If parity is enabled, just read it and ignore it. */
221  /* TODO: Should we check parity? Or is it too slow for nothing added..*/
222  if (arg->parity_ == UART_CONFIG_PARITY_EVEN)
223  arg->read_bit_(&wait, start);
224  else if (arg->parity_ == UART_CONFIG_PARITY_ODD)
225  arg->read_bit_(&wait, start);
226 
227  // Stop bit
228  arg->wait_(&wait, start);
229  if (arg->stop_bits_ == 2)
230  arg->wait_(&wait, start);
231 
232  arg->rx_buffer_[arg->rx_in_pos_] = rec;
233  arg->rx_in_pos_ = (arg->rx_in_pos_ + 1) % arg->rx_buffer_size_;
234  // Clear RX pin so that the interrupt doesn't re-trigger right away again.
235  arg->rx_pin_->clear_interrupt();

237 void ICACHE_RAM_ATTR HOT ESP8266SoftwareSerial::write_byte(uint8_t data) {
238  if (this->tx_pin_ == nullptr) {
239  ESP_LOGE(TAG, "UART doesn't have TX pins set!");
240  return;
241  }
242  bool parity_bit = false;
243  bool need_parity_bit = true;
244  if (this->parity_ == UART_CONFIG_PARITY_EVEN)
245  parity_bit = true;
246  else if (this->parity_ == UART_CONFIG_PARITY_ODD)
247  parity_bit = false;
248  else
249  need_parity_bit = false;
250 
251  {
252  InterruptLock lock;
253  uint32_t wait = this->bit_time_;
254  const uint32_t start = ESP.getCycleCount();
255  // Start bit
256  this->write_bit_(false, &wait, start);
257  for (int i = 0; i < this->data_bits_; i++) {
258  bool bit = data & (1 << i);
259  this->write_bit_(bit, &wait, start);
260  if (need_parity_bit)
261  parity_bit ^= bit;
262  }
263  if (need_parity_bit)
264  this->write_bit_(parity_bit, &wait, start);
265  // Stop bit
266  this->write_bit_(true, &wait, start);
267  if (this->stop_bits_ == 2)
268  this->wait_(&wait, start);
269  }
