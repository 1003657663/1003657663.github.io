50 void Sim800LComponent::parse_cmd_(std::string message) {
51  ESP_LOGV(TAG, "R: %s - %d", message.c_str(), this->state_);
52 
53  if (message.empty())
54  return;
55 
56  if (this->expect_ack_) {
57  bool ok = message == "OK";
58  this->expect_ack_ = false;
59  if (!ok) {
60  if (this->state_ == STATE_CHECK_AT && message == "AT") {
61  // Expected ack but AT echo received
62  this->state_ = STATE_DISABLE_ECHO;
63  this->expect_ack_ = true;
64  } else {
65  ESP_LOGW(TAG, "Not ack. %d %s", this->state_, message.c_str());
66  this->state_ = STATE_IDLE; // Let it timeout
67  return;
68  }
69  }
70  }
71 
72  switch (this->state_) {
73  case STATE_INIT: {
74  // While we were waiting for update to check for messages, this notifies a message
75  // is available.
76  bool message_available = message.compare(0, 6, "+CMTI:") == 0;
77  if (!message_available)
78  break;
79  // Else fall thru ...
80  }
81  case STATE_CHECK_SMS:
82  send_cmd_("AT+CMGL=\"ALL\"");
83  this->state_ = STATE_PARSE_SMS;
84  this->parse_index_ = 0;
85  break;
86  case STATE_DISABLE_ECHO:
87  send_cmd_("ATE0");
88  this->state_ = STATE_CHECK_AT;
89  this->expect_ack_ = true;
90  break;
91  case STATE_CHECK_AT:
92  send_cmd_("AT+CMGF=1");
93  this->state_ = STATE_CREG;
94  this->expect_ack_ = true;
95  break;
96  case STATE_CREG:
97  send_cmd_("AT+CREG?");
98  this->state_ = STATE_CREGWAIT;
99  break;
100  case STATE_CREGWAIT: {
101  // Response: "+CREG: 0,1" -- the one there means registered ok
102  // "+CREG: -,-" means not registered ok
103  bool registered = message.compare(0, 6, "+CREG:") == 0 && (message[9] == '1' || message[9] == '5');
104  if (registered) {
105  if (!this->registered_)
106  ESP_LOGD(TAG, "Registered OK");
107  this->state_ = STATE_CSQ;
108  this->expect_ack_ = true;
109  } else {
110  ESP_LOGW(TAG, "Registration Fail");
111  if (message[7] == '0') { // Network registration is disable, enable it
112  send_cmd_("AT+CREG=1");
113  this->expect_ack_ = true;
114  this->state_ = STATE_CHECK_AT;
115  } else {
116  // Keep waiting registration
117  this->state_ = STATE_CREG;
118  }
119  }
120  this->registered_ = registered;
121  break;
122  }
123  case STATE_CSQ:
124  send_cmd_("AT+CSQ");
125  this->state_ = STATE_CSQ_RESPONSE;
126  break;
127  case STATE_CSQ_RESPONSE:
128  if (message.compare(0, 5, "+CSQ:") == 0) {
129  size_t comma = message.find(',', 6);
130  if (comma != 6) {
131  this->rssi_ = strtol(message.substr(6, comma - 6).c_str(), nullptr, 10);
132  ESP_LOGD(TAG, "RSSI: %d", this->rssi_);
133  }
134  }
135  this->expect_ack_ = true;
136  this->state_ = STATE_CHECK_SMS;
137  break;
138  case STATE_PARSE_SMS:
139  this->state_ = STATE_PARSE_SMS_RESPONSE;
140  break;
141  case STATE_PARSE_SMS_RESPONSE:
142  if (message.compare(0, 6, "+CMGL:") == 0 && this->parse_index_ == 0) {
143  size_t start = 7;
144  size_t end = message.find(',', start);
145  uint8_t item = 0;
146  while (end != start) {
147  item++;
148  if (item == 1) { // Slot Index
149  this->parse_index_ = strtol(message.substr(start, end - start).c_str(), nullptr, 10);
150  }
151  // item 2 = STATUS, usually "REC UNERAD"
152  if (item == 3) { // recipient
153  // Add 1 and remove 2 from substring to get rid of "quotes"
154  this->sender_ = message.substr(start + 1, end - start - 2);
155  break;
156  }
157  // item 4 = ""
158  // item 5 = Received timestamp
159  start = end + 1;
160  end = message.find(',', start);
161  }
162 
163  if (item < 2) {
164  ESP_LOGD(TAG, "Invalid message %d %s", this->state_, message.c_str());
165  return;
166  }
167  this->state_ = STATE_RECEIVESMS;
168  }
169  // Otherwise we receive another OK, we do nothing just wait polling to continuously check for SMS
170  if (message == "OK")
171  this->state_ = STATE_INIT;
172  break;
173  case STATE_RECEIVESMS:
174  /* Our recipient is set and the message body is in message
175  kick ESPHome callback now
176  */
177  ESP_LOGD(TAG, "Received SMS from: %s", this->sender_.c_str());
178  ESP_LOGD(TAG, "%s", message.c_str());
179  this->callback_.call(message, this->sender_);
180  /* If the message is multiline, next lines will contain message data.
181  If there were other messages in the list, next line will be +CMGL: ...
182  At the end of the list the new line and the OK should be received.
183  To keep this simple just first line of message if considered, then
184  the next state will swallow all received data and in next poll event
185  this message index is marked for deletion.
186  */
187  this->state_ = STATE_RECEIVEDSMS;
188  break;
189  case STATE_RECEIVEDSMS:
190  // Let the buffer flush. Next poll will request to delete the parsed index message.
191  break;
192  case STATE_SENDINGSMS1:
193  this->send_cmd_("AT+CMGS=\"" + this->recipient_ + "\"");
194  this->state_ = STATE_SENDINGSMS2;
195  break;
196  case STATE_SENDINGSMS2:
197  if (message == ">") {
198  // Send sms body
199  ESP_LOGD(TAG, "Sending message: '%s'", this->outgoing_message_.c_str());
200  this->write_str(this->outgoing_message_.c_str());
201  this->write(26);
202  this->state_ = STATE_SENDINGSMS3;
203  } else {
204  this->registered_ = false;
205  this->state_ = STATE_INIT;
206  this->send_cmd_("AT+CMEE=2");
207  this->write(26);
208  }
209  break;
210  case STATE_SENDINGSMS3:
211  if (message.compare(0, 6, "+CMGS:") == 0) {
212  ESP_LOGD(TAG, "SMS Sent OK: %s", message.c_str());
213  this->send_pending_ = false;
214  this->state_ = STATE_CHECK_SMS;
215  this->expect_ack_ = true;
216  }
217  break;
218  case STATE_DIALING1:
219  this->send_cmd_("ATD" + this->recipient_ + ';');
220  this->state_ = STATE_DIALING2;
221  break;
222  case STATE_DIALING2:
223  if (message == "OK") {
224  // Dialing
225  ESP_LOGD(TAG, "Dialing: '%s'", this->recipient_.c_str());
226  this->state_ = STATE_INIT;
227  this->dial_pending_ = false;
228  } else {
229  this->registered_ = false;
230  this->state_ = STATE_INIT;
231  this->send_cmd_("AT+CMEE=2");
232  this->write(26);
233  }
234  break;

241 void Sim800LComponent::loop() {
242  // Read message
243  while (this->available()) {
244  uint8_t byte;
245  this->read_byte(&byte);
246 
247  if (this->read_pos_ == SIM800L_READ_BUFFER_LENGTH)
248  this->read_pos_ = 0;
249 
250  ESP_LOGVV(TAG, "Buffer pos: %u %d", this->read_pos_, byte); // NOLINT
251 
252  if (byte == ASCII_CR)
253  continue;
254  if (byte >= 0x7F)
255  byte = '?'; // need to be valid utf8 string for log functions.
256  this->read_buffer_[this->read_pos_] = byte;
257 
258  if (this->state_ == STATE_SENDINGSMS2 && this->read_pos_ == 0 && byte == '>')
259  this->read_buffer_[++this->read_pos_] = ASCII_LF;
260 
261  if (this->read_buffer_[this->read_pos_] == ASCII_LF) {
262  this->read_buffer_[this->read_pos_] = 0;
263  this->read_pos_ = 0;
264  this->parse_cmd_(this->read_buffer_);
265  } else {
266  this->read_pos_++;
267  }
268  }
