124  // Enable sensor

126  uint8_t meas_register = 0;
127  meas_register |= (this->temperature_oversampling_ & 0b111) << 5;
128  meas_register |= (this->pressure_oversampling_ & 0b111) << 2;
129  meas_register |= 0b01; // Forced mode
130  if (!this->write_byte(BMP280_REGISTER_CONTROL, meas_register)) {
131  this->status_set_warning();
132  return;
133  }
134 
135  float meas_time = 1;
136  meas_time += 2.3f * oversampling_to_time(this->temperature_oversampling_);
137  meas_time += 2.3f * oversampling_to_time(this->pressure_oversampling_) + 0.575f;
138 
139  this->set_timeout("data", uint32_t(ceilf(meas_time)), [this]() {
140  int32_t t_fine = 0;
141  float temperature = this->read_temperature_(&t_fine);
142  if (isnan(temperature)) {
143  ESP_LOGW(TAG, "Invalid temperature, cannot read pressure values.");
144  this->status_set_warning();
145  return;
146  }
147  float pressure = this->read_pressure_(t_fine);
148 
149  ESP_LOGD(TAG, "Got temperature=%.1f°C pressure=%.1fhPa", temperature, pressure);
150  if (this->temperature_sensor_ != nullptr)
151  this->temperature_sensor_->publish_state(temperature);
152  if (this->pressure_sensor_ != nullptr)
153  this->pressure_sensor_->publish_state(pressure);
154  this->status_clear_warning();

158 float BMP280Component::read_temperature_(int32_t *t_fine) {
159  uint8_t data[3];
160  if (!this->read_bytes(BMP280_REGISTER_TEMPDATA, data, 3))
161  return NAN;
162  int32_t adc = ((data[0] & 0xFF) << 16) | ((data[1] & 0xFF) << 8) | (data[2] & 0xFF);
163  adc >>= 4;
164  if (adc == 0x80000)
165  // temperature was disabled
166  return NAN;
167 
168  const int32_t t1 = this->calibration_.t1;
169  const int32_t t2 = this->calibration_.t2;
170  const int32_t t3 = this->calibration_.t3;
171 
172  int32_t var1 = (((adc >> 3) - (t1 << 1)) * t2) >> 11;
173  int32_t var2 = (((((adc >> 4) - t1) * ((adc >> 4) - t1)) >> 12) * t3) >> 14;
174  *t_fine = var1 + var2;
175 
176  float temperature = (*t_fine * 5 + 128) >> 8;
177  return temperature / 100.0f;
178 }

180 float BMP280Component::read_pressure_(int32_t t_fine) {
181  uint8_t data[3];
182  if (!this->read_bytes(BMP280_REGISTER_PRESSUREDATA, data, 3))
183  return NAN;
184  int32_t adc = ((data[0] & 0xFF) << 16) | ((data[1] & 0xFF) << 8) | (data[2] & 0xFF);
185  adc >>= 4;
186  if (adc == 0x80000)
187  // pressure was disabled
188  return NAN;
189  const int64_t p1 = this->calibration_.p1;
190  const int64_t p2 = this->calibration_.p2;
191  const int64_t p3 = this->calibration_.p3;
192  const int64_t p4 = this->calibration_.p4;
193  const int64_t p5 = this->calibration_.p5;
194  const int64_t p6 = this->calibration_.p6;
195  const int64_t p7 = this->calibration_.p7;
196  const int64_t p8 = this->calibration_.p8;
197  const int64_t p9 = this->calibration_.p9;
198 
199  int64_t var1, var2, p;
200  var1 = int64_t(t_fine) - 128000;
201  var2 = var1 * var1 * p6;
202  var2 = var2 + ((var1 * p5) << 17);
203  var2 = var2 + (p4 << 35);
204  var1 = ((var1 * var1 * p3) >> 8) + ((var1 * p2) << 12);
205  var1 = ((int64_t(1) << 47) + var1) * p1 >> 33;
206 
207  if (var1 == 0)
208  return NAN;
209 
210  p = 1048576 - adc;
211  p = (((p << 31) - var2) * 3125) / var1;
212  var1 = (p9 * (p >> 13) * (p >> 13)) >> 25;
213  var2 = (p8 * p) >> 19;
214 
215  p = ((p + var1 + var2) >> 8) + (p7 << 4);
216  return (p / 256.0f) / 100.0f;
