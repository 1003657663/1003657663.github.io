124 void ESP32Camera::loop() {
125  // check if we can return the image
126  if (this->can_return_image_()) {
127  // return image
128  auto *fb = this->current_image_->get_raw_buffer();
129  xQueueSend(this->framebuffer_return_queue_, &fb, portMAX_DELAY);
130  this->current_image_.reset();
131  }
132 
133  // Check if we should fetch a new image
134  if (!this->has_requested_image_())
135  return;
136  if (this->current_image_.use_count() > 1) {
137  // image is still in use
138  return;
139  }
140  const uint32_t now = millis();
141  if (now - this->last_update_ <= this->max_update_interval_)
142  return;
143 
144  // request new image
145  camera_fb_t *fb;
146  if (xQueueReceive(this->framebuffer_get_queue_, &fb, 0L) != pdTRUE) {
147  // no frame ready
148  ESP_LOGVV(TAG, "No frame ready");
149  return;
150  }
151 
152  if (fb == nullptr) {
153  ESP_LOGW(TAG, "Got invalid frame from camera!");
154  xQueueSend(this->framebuffer_return_queue_, &fb, portMAX_DELAY);
155  return;
156  }
157  this->current_image_ = std::make_shared<CameraImage>(fb);
158 
159  ESP_LOGD(TAG, "Got Image: len=%u", fb->len);
160  this->new_image_callback_.call(this->current_image_);
161  this->last_update_ = now;
162  this->single_requester_ = false;
163 }
