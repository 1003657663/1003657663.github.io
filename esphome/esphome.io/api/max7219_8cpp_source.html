170 uint8_t MAX7219Component::print(uint8_t start_pos, const char *str) {
171  uint8_t pos = start_pos;
172  for (; *str != '\0'; str++) {
173  uint8_t data = MAX7219_UNKNOWN_CHAR;
174  if (*str >= ' ' && *str <= '~')
175  data = pgm_read_byte(&MAX7219_ASCII_TO_RAW[*str - ' ']);
176 
177  if (data == MAX7219_UNKNOWN_CHAR) {
178  ESP_LOGW(TAG, "Encountered character '%c' with no MAX7219 representation while translating string!", *str);
179  }
180  if (*str == '.') {
181  if (pos != start_pos)
182  pos--;
183  this->buffer_[pos] |= 0b10000000;
184  } else {
185  if (pos >= this->num_chips_ * 8) {
186  ESP_LOGE(TAG, "MAX7219 String is too long for the display!");
187  break;
188  }
189  this->buffer_[pos] = data;
190  }
191  pos++;
192  }
193  return pos - start_pos;
