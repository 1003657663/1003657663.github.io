25 uint32_t ICACHE_RAM_ATTR HOT AcDimmerDataStore::timer_intr(uint32_t now) {
26  // If no ZC signal received yet.
27  if (this->crossed_zero_at == 0)
28  return 0;
29 
30  uint32_t time_since_zc = now - this->crossed_zero_at;
31  if (this->value == 65535 || this->value == 0) {
32  return 0;
33  }
34 
35  if (this->enable_time_us != 0 && time_since_zc >= this->enable_time_us) {
36  this->enable_time_us = 0;
37  this->gate_pin->digital_write(true);
38  // Prevent too short pulses
39  this->disable_time_us = max(this->disable_time_us, time_since_zc + GATE_ENABLE_TIME);
40  }
41  if (this->disable_time_us != 0 && time_since_zc >= this->disable_time_us) {
42  this->disable_time_us = 0;
43  this->gate_pin->digital_write(false);
44  }
45 
46  if (time_since_zc < this->enable_time_us)
47  // Next event is enable, return time until that event
48  return this->enable_time_us - time_since_zc;
49  else if (time_since_zc < disable_time_us) {
50  // Next event is disable, return time until that event
51  return this->disable_time_us - time_since_zc;
52  }
53 
54  if (time_since_zc >= this->cycle_time_us) {
55  // Already past last cycle time, schedule next call shortly
56  return 100;
57  }
58 
59  return this->cycle_time_us - time_since_zc;
60 }

63 uint32_t ICACHE_RAM_ATTR HOT timer_interrupt() {
64  // run at least with 1kHz
65  uint32_t min_dt_us = 1000;
66  uint32_t now = micros();
67  for (auto *dimmer : all_dimmers) {
68  if (dimmer == nullptr)
69  // no more dimmers
70  break;
71  uint32_t res = dimmer->timer_intr(now);
72  if (res != 0 && res < min_dt_us)
73  min_dt_us = res;
74  }

80 void ICACHE_RAM_ATTR HOT AcDimmerDataStore::gpio_intr() {
81  uint32_t prev_crossed = this->crossed_zero_at;
82 
83  // 50Hz mains frequency should give a half cycle of 10ms a 60Hz will give 8.33ms
84  // in any case the cycle last at least 5ms
85  this->crossed_zero_at = micros();
86  uint32_t cycle_time = this->crossed_zero_at - prev_crossed;
87  if (cycle_time > 5000) {
88  this->cycle_time_us = cycle_time;
89  } else {
90  // Otherwise this is noise and this is 2nd (or 3rd...) fall in the same pulse
91  // Consider this is the right fall edge and accumulate the cycle time instead
92  this->cycle_time_us += cycle_time;
93  }
94 
95  if (this->value == 65535) {
96  // fully on, enable output immediately
97  this->gate_pin->digital_write(true);
98  } else if (this->init_cycle) {
99  // send a full cycle
100  this->init_cycle = false;
101  this->enable_time_us = 0;
102  this->disable_time_us = cycle_time_us;
103  } else if (this->value == 0) {
104  // fully off, disable output immediately
105  this->gate_pin->digital_write(false);
106  } else {
107  if (this->method == DIM_METHOD_TRAILING) {
108  this->enable_time_us = 1; // cannot be 0
109  this->disable_time_us = max((uint32_t) 10, this->value * this->cycle_time_us / 65535);
110  } else {
111  // calculate time until enable in µs: (1.0-value)*cycle_time, but with integer arithmetic
112  // also take into account min_power
113  auto min_us = this->cycle_time_us * this->min_power / 1000;
114  this->enable_time_us = max((uint32_t) 1, ((65535 - this->value) * (this->cycle_time_us - min_us)) / 65535);
115  if (this->method == DIM_METHOD_LEADING_PULSE) {
116  // Minimum pulse time should be enough for the triac to trigger when it is close to the ZC zone
117  // this is for brightness near 99%
118  this->disable_time_us = max(this->enable_time_us + GATE_ENABLE_TIME, (uint32_t) cycle_time_us / 10);
119  } else {
120  this->gate_pin->digital_write(false);
121  this->disable_time_us = this->cycle_time_us;
122  }
123  }
124  }

127 void ICACHE_RAM_ATTR HOT AcDimmerDataStore::s_gpio_intr(AcDimmerDataStore *store) {
128  // Attaching pin interrupts on the same pin will override the previous interupt
129  // However, the user expects that multiple dimmers sharing the same ZC pin will work.
130  // We solve this in a bit of a hacky way: On each pin interrupt, we check all dimmers
131  // if any of them are using the same ZC pin, and also trigger the interrupt for *them*.
132  for (auto *dimmer : all_dimmers) {
133  if (dimmer == nullptr)
134  break;
135  if (dimmer->zero_cross_pin_number == store->zero_cross_pin_number) {
136  dimmer->gpio_intr();
137  }
138  }
139 }
