52 void BangBangClimate::compute_state_() {
53  if (this->mode != climate::CLIMATE_MODE_AUTO) {
54  // in non-auto mode, switch directly to appropriate action
55  // - HEAT mode -> HEATING action
56  // - COOL mode -> COOLING action
57  // - OFF mode -> OFF action (not IDLE!)
58  this->switch_to_action_(static_cast<climate::ClimateAction>(this->mode));
59  return;
60  }
61  if (isnan(this->current_temperature) || isnan(this->target_temperature_low) || isnan(this->target_temperature_high)) {
62  // if any control parameters are nan, go to OFF action (not IDLE!)
63  this->switch_to_action_(climate::CLIMATE_ACTION_OFF);
64  return;
65  }
66  const bool too_cold = this->current_temperature < this->target_temperature_low;
67  const bool too_hot = this->current_temperature > this->target_temperature_high;
68 
69  climate::ClimateAction target_action;
70  if (too_cold) {
71  // too cold -> enable heating if possible, else idle
72  if (this->supports_heat_)
73  target_action = climate::CLIMATE_ACTION_HEATING;
74  else
75  target_action = climate::CLIMATE_ACTION_IDLE;
76  } else if (too_hot) {
77  // too hot -> enable cooling if possible, else idle
78  if (this->supports_cool_)
79  target_action = climate::CLIMATE_ACTION_COOLING;
80  else
81  target_action = climate::CLIMATE_ACTION_IDLE;
82  } else {
83  // neither too hot nor too cold -> in range
84  if (this->supports_cool_ && this->supports_heat_) {
85  // if supports both ends, go to idle action
86  target_action = climate::CLIMATE_ACTION_IDLE;
87  } else {
88  // else use current mode and don't change (hysteresis)
89  target_action = this->action;
90  }
91  }
92 
93  this->switch_to_action_(target_action);
94 }
95 void BangBangClimate::switch_to_action_(climate::ClimateAction action) {
96  if (action == this->action)
97  // already in target mode
98  return;
99 
100  if ((action == climate::CLIMATE_ACTION_OFF && this->action == climate::CLIMATE_ACTION_IDLE) ||
101  (action == climate::CLIMATE_ACTION_IDLE && this->action == climate::CLIMATE_ACTION_OFF)) {
102  // switching from OFF to IDLE or vice-versa
103  // these only have visual difference. OFF means user manually disabled,
104  // IDLE means it's in auto mode but value is in target range.
105  this->action = action;
106  this->publish_state();
107  return;
108  }
109 
110  if (this->prev_trigger_ != nullptr) {
111  this->prev_trigger_->stop_action();
112  this->prev_trigger_ = nullptr;
113  }
114  Trigger<> *trig;
115  switch (action) {
116  case climate::CLIMATE_ACTION_OFF:
117  case climate::CLIMATE_ACTION_IDLE:
118  trig = this->idle_trigger_;
119  break;
120  case climate::CLIMATE_ACTION_COOLING:
121  trig = this->cool_trigger_;
122  break;
123  case climate::CLIMATE_ACTION_HEATING:
124  trig = this->heat_trigger_;
125  break;
126  default:
127  trig = nullptr;
128  }
129  assert(trig != nullptr);
130  trig->trigger();
131  this->action = action;
132  this->prev_trigger_ = trig;
133  this->publish_state();
134 }
