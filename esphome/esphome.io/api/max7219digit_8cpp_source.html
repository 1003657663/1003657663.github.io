59 
60  // check if the buffer has shrunk past the current position since last update
61  if ((this->max_displaybuffer_.size() >= this->old_buffer_size_ + 3) ||
62  (this->max_displaybuffer_.size() <= this->old_buffer_size_ - 3)) {
63  this->stepsleft_ = 0;

65  this->old_buffer_size_ = this->max_displaybuffer_.size();
66  }
67 
68  // Reset the counter back to 0 when full string has been displayed.
69  if (this->stepsleft_ > this->max_displaybuffer_.size())
70  this->stepsleft_ = 0;
71 
72  // Return if there is no need to scroll or scroll is off
73  if (!this->scroll_ || (this->max_displaybuffer_.size() <= this->num_chips_ * 8)) {
74  this->display();
75  return;
76  }
77 
78  if ((this->stepsleft_ == 0) && (now - this->last_scroll_ < this->scroll_delay_)) {
79  this->display();
80  return;
81  }
82 
83  // Dwell time at end of string in case of stop at end
84  if (this->scroll_mode_ == 1) {

87  this->stepsleft_ = 0;

89  this->display();

91  return;
92  }
93  }
94 
95  // Actual call to scroll left action
96  if (now - this->last_scroll_ >= this->scroll_speed_) {

220 void MAX7219Component::send64pixels(uint8_t chip, const uint8_t pixels[8]) {
221  for (uint8_t col = 0; col < 8; col++) { // RUN THIS LOOP 8 times until column is 7
222  this->enable(); // start sending by enabling SPI
223  for (uint8_t i = 0; i < chip; i++) // send extra NOPs to push the pixels out to extra displays
224  this->send_byte_(MAX7219_REGISTER_NOOP,
225  MAX7219_REGISTER_NOOP); // run this loop unit the matching chip is reached
226  uint8_t b = 0; // rotate pixels 90 degrees -- set byte to 0
227  if (this->orientation_ == 0) {
228  for (uint8_t i = 0; i < 8; i++) {
229  // run this loop 8 times for all the pixels[8] received
230  b |= ((pixels[i] >> col) & 1) << (7 - i); // change the column bits into row bits
231  }
232  } else if (this->orientation_ == 1) {
233  b = pixels[col];
234  } else if (this->orientation_ == 2) {
235  for (uint8_t i = 0; i < 8; i++) {
236  b |= ((pixels[i] >> (7 - col)) & 1) << i;
237  }
238  } else {
239  b = pixels[7 - col];
240  }
241  // send this byte to dispay at selected chip
242  if (this->invert_) {
243  this->send_byte_(col + 1, ~b);
244  } else {
245  this->send_byte_(col + 1, b);
246  }
247  for (int i = 0; i < this->num_chips_ - chip - 1; i++) // end with enough NOPs so later chips don't update
248  this->send_byte_(MAX7219_REGISTER_NOOP, MAX7219_REGISTER_NOOP);
249  this->disable(); // all done disable SPI
250  } // end of for each column
251 } // end of send64pixels
