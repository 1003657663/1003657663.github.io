49  void calculate_gamma_table(float gamma);
50  inline Color color_correct(Color color) const ALWAYS_INLINE {
51  // corrected = (uncorrected * max_brightness * local_brightness) ^ gamma
52  return Color(this->color_correct_red(color.red), this->color_correct_green(color.green),
53  this->color_correct_blue(color.blue), this->color_correct_white(color.white));
54  }
55  inline uint8_t color_correct_red(uint8_t red) const ALWAYS_INLINE {
56  uint8_t res = esp_scale8(esp_scale8(red, this->max_brightness_.red), this->local_brightness_);
57  return this->gamma_table_[res];
58  }
59  inline uint8_t color_correct_green(uint8_t green) const ALWAYS_INLINE {
60  uint8_t res = esp_scale8(esp_scale8(green, this->max_brightness_.green), this->local_brightness_);
61  return this->gamma_table_[res];
62  }
63  inline uint8_t color_correct_blue(uint8_t blue) const ALWAYS_INLINE {
64  uint8_t res = esp_scale8(esp_scale8(blue, this->max_brightness_.blue), this->local_brightness_);
65  return this->gamma_table_[res];
66  }
67  inline uint8_t color_correct_white(uint8_t white) const ALWAYS_INLINE {
68  // do not scale white value with brightness
69  uint8_t res = esp_scale8(white, this->max_brightness_.white);
70  return this->gamma_table_[res];
71  }
72  inline Color color_uncorrect(Color color) const ALWAYS_INLINE {
73  // uncorrected = corrected^(1/gamma) / (max_brightness * local_brightness)
74  return Color(this->color_uncorrect_red(color.red), this->color_uncorrect_green(color.green),
75  this->color_uncorrect_blue(color.blue), this->color_uncorrect_white(color.white));
76  }
77  inline uint8_t color_uncorrect_red(uint8_t red) const ALWAYS_INLINE {
78  if (this->max_brightness_.red == 0 || this->local_brightness_ == 0)
79  return 0;
80  uint16_t uncorrected = this->gamma_reverse_table_[red] * 255UL;
81  uint8_t res = ((uncorrected / this->max_brightness_.red) * 255UL) / this->local_brightness_;
82  return res;
83  }
84  inline uint8_t color_uncorrect_green(uint8_t green) const ALWAYS_INLINE {
85  if (this->max_brightness_.green == 0 || this->local_brightness_ == 0)
86  return 0;
87  uint16_t uncorrected = this->gamma_reverse_table_[green] * 255UL;
88  uint8_t res = ((uncorrected / this->max_brightness_.green) * 255UL) / this->local_brightness_;
89  return res;
90  }
91  inline uint8_t color_uncorrect_blue(uint8_t blue) const ALWAYS_INLINE {
92  if (this->max_brightness_.blue == 0 || this->local_brightness_ == 0)
93  return 0;
94  uint16_t uncorrected = this->gamma_reverse_table_[blue] * 255UL;
95  uint8_t res = ((uncorrected / this->max_brightness_.blue) * 255UL) / this->local_brightness_;
96  return res;
97  }
98  inline uint8_t color_uncorrect_white(uint8_t white) const ALWAYS_INLINE {
99  if (this->max_brightness_.white == 0)
100  return 0;
101  uint16_t uncorrected = this->gamma_reverse_table_[white] * 255UL;
102  uint8_t res = uncorrected / this->max_brightness_.white;
103  return res;
104  }
