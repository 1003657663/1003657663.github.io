131  enum PcdCommand : uint8_t {

133  PCD_MEM = 0x01, // stores 25 uint8_ts into the internal buffer

156  PICC_CMD_SEL_CL3 = 0x97, // Anti collision/Select, Cascade Level 3

158  PICC_CMD_RATS = 0xE0, // Request command for Answer To Reset.

160  // Use PCD_MFAuthent to authenticate access to a sector, then use these commands to read/write/modify the blocks on
