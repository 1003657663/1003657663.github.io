9 void ThermostatClimate::setup() {
10  this->sensor_->add_on_state_callback([this](float state) {
11  this->current_temperature = state;
12  // required action may have changed, recompute, refresh
13  this->switch_to_action_(compute_action_());
14  // current temperature and possibly action changed, so publish the new state
15  this->publish_state();
16  });
17  this->current_temperature = this->sensor_->state;
18  // restore all climate data, if possible
19  auto restore = this->restore_state_();
20  if (restore.has_value()) {
21  restore->to_call(this).perform();
22  } else {
23  // restore from defaults, change_away handles temps for us
24  this->mode = climate::CLIMATE_MODE_AUTO;
25  this->change_away_(false);
26  }
27  // refresh the climate action based on the restored settings
28  this->switch_to_action_(compute_action_());
29  this->setup_complete_ = true;
30  this->publish_state();
31 }

104 climate::ClimateAction ThermostatClimate::compute_action_() {
105  climate::ClimateAction target_action = this->action;
106  if (this->supports_two_points_) {
107  if (isnan(this->current_temperature) || isnan(this->target_temperature_low) ||
108  isnan(this->target_temperature_high) || isnan(this->hysteresis_))
109  // if any control parameters are nan, go to OFF action (not IDLE!)
110  return climate::CLIMATE_ACTION_OFF;
111 
112  if (((this->action == climate::CLIMATE_ACTION_FAN) && (this->mode != climate::CLIMATE_MODE_FAN_ONLY)) ||
113  ((this->action == climate::CLIMATE_ACTION_DRYING) && (this->mode != climate::CLIMATE_MODE_DRY))) {
114  target_action = climate::CLIMATE_ACTION_IDLE;
115  }
116 
117  switch (this->mode) {
118  case climate::CLIMATE_MODE_FAN_ONLY:
119  if (this->supports_fan_only_) {
120  if (this->current_temperature > this->target_temperature_high + this->hysteresis_)
121  target_action = climate::CLIMATE_ACTION_FAN;
122  else if (this->current_temperature < this->target_temperature_high - this->hysteresis_)
123  if (this->action == climate::CLIMATE_ACTION_FAN)
124  target_action = climate::CLIMATE_ACTION_IDLE;
125  }
126  break;
127  case climate::CLIMATE_MODE_DRY:
128  target_action = climate::CLIMATE_ACTION_DRYING;
129  break;
130  case climate::CLIMATE_MODE_OFF:
131  target_action = climate::CLIMATE_ACTION_OFF;
132  break;
133  case climate::CLIMATE_MODE_AUTO:
134  case climate::CLIMATE_MODE_COOL:
135  case climate::CLIMATE_MODE_HEAT:
136  if (this->supports_cool_) {
137  if (this->current_temperature > this->target_temperature_high + this->hysteresis_)
138  target_action = climate::CLIMATE_ACTION_COOLING;
139  else if (this->current_temperature < this->target_temperature_high - this->hysteresis_)
140  if (this->action == climate::CLIMATE_ACTION_COOLING)
141  target_action = climate::CLIMATE_ACTION_IDLE;
142  }
143  if (this->supports_heat_) {
144  if (this->current_temperature < this->target_temperature_low - this->hysteresis_)
145  target_action = climate::CLIMATE_ACTION_HEATING;
146  else if (this->current_temperature > this->target_temperature_low + this->hysteresis_)
147  if (this->action == climate::CLIMATE_ACTION_HEATING)
148  target_action = climate::CLIMATE_ACTION_IDLE;
149  }
150  break;
151  default:
152  break;
153  }
154  } else {
155  if (isnan(this->current_temperature) || isnan(this->target_temperature) || isnan(this->hysteresis_))
156  // if any control parameters are nan, go to OFF action (not IDLE!)
157  return climate::CLIMATE_ACTION_OFF;

161  target_action = climate::CLIMATE_ACTION_IDLE;
162  }

173  break;
174  case climate::CLIMATE_MODE_DRY:
175  target_action = climate::CLIMATE_ACTION_DRYING;
176  break;
177  case climate::CLIMATE_MODE_OFF:
178  target_action = climate::CLIMATE_ACTION_OFF;
179  break;
180  case climate::CLIMATE_MODE_COOL:
181  if (this->supports_cool_) {
182  if (this->current_temperature > this->target_temperature + this->hysteresis_)
183  target_action = climate::CLIMATE_ACTION_COOLING;
184  else if (this->current_temperature < this->target_temperature - this->hysteresis_)
185  if (this->action == climate::CLIMATE_ACTION_COOLING)
186  target_action = climate::CLIMATE_ACTION_IDLE;
187  }
188  case climate::CLIMATE_MODE_HEAT:

209 void ThermostatClimate::switch_to_action_(climate::ClimateAction action) {
210  // setup_complete_ helps us ensure an action is called immediately after boot
211  if ((action == this->action) && this->setup_complete_)
212  // already in target mode
213  return;
214 
215  if (((action == climate::CLIMATE_ACTION_OFF && this->action == climate::CLIMATE_ACTION_IDLE) ||
216  (action == climate::CLIMATE_ACTION_IDLE && this->action == climate::CLIMATE_ACTION_OFF)) &&
217  this->setup_complete_) {
218  // switching from OFF to IDLE or vice-versa
219  // these only have visual difference. OFF means user manually disabled,
220  // IDLE means it's in auto mode but value is in target range.
221  this->action = action;
222  return;
223  }
224 
225  if (this->prev_action_trigger_ != nullptr) {
226  this->prev_action_trigger_->stop_action();
227  this->prev_action_trigger_ = nullptr;
228  }
229  Trigger<> *trig = this->idle_action_trigger_;
230  switch (action) {
231  case climate::CLIMATE_ACTION_OFF:
232  case climate::CLIMATE_ACTION_IDLE:
233  // trig = this->idle_action_trigger_;
234  break;
235  case climate::CLIMATE_ACTION_COOLING:
236  trig = this->cool_action_trigger_;
237  break;
238  case climate::CLIMATE_ACTION_HEATING:
239  trig = this->heat_action_trigger_;
240  break;
241  case climate::CLIMATE_ACTION_FAN:
242  trig = this->fan_only_action_trigger_;
243  break;
244  case climate::CLIMATE_ACTION_DRYING:
245  trig = this->dry_action_trigger_;
246  break;
247  default:
248  // we cannot report an invalid mode back to HA (even if it asked for one)
249  // and must assume some valid value
250  action = climate::CLIMATE_ACTION_OFF;
251  // trig = this->idle_action_trigger_;
252  }
253  assert(trig != nullptr);
254  trig->trigger();
255  this->action = action;
256  this->prev_action_trigger_ = trig;
257 }
