54 bool Tuya::validate_message_() {
55  uint32_t at = this->rx_message_.size() - 1;
56  auto *data = &this->rx_message_[0];
57  uint8_t new_byte = data[at];
58 
59  // Byte 0: HEADER1 (always 0x55)
60  if (at == 0)
61  return new_byte == 0x55;
62  // Byte 1: HEADER2 (always 0xAA)
63  if (at == 1)
64  return new_byte == 0xAA;
65 
66  // Byte 2: VERSION
67  // no validation for the following fields:
68  uint8_t version = data[2];
69  if (at == 2)
70  return true;
71  // Byte 3: COMMAND
72  uint8_t command = data[3];
73  if (at == 3)
74  return true;
75 
76  // Byte 4: LENGTH1
77  // Byte 5: LENGTH2
78  if (at <= 5)
79  // no validation for these fields
80  return true;
81 
82  uint16_t length = (uint16_t(data[4]) << 8) | (uint16_t(data[5]));
83 
84  // wait until all data is read
85  if (at - 6 < length)
86  return true;
87 
88  // Byte 6+LEN: CHECKSUM - sum of all bytes (including header) modulo 256
89  uint8_t rx_checksum = new_byte;
90  uint8_t calc_checksum = 0;
91  for (uint32_t i = 0; i < 6 + length; i++)
92  calc_checksum += data[i];
93 
94  if (rx_checksum != calc_checksum) {
95  ESP_LOGW(TAG, "Tuya Received invalid message checksum %02X!=%02X", rx_checksum, calc_checksum);
96  return false;
97  }
98 
99  // valid message
100  const uint8_t *message_data = data + 6;
101  ESP_LOGV(TAG, "Received Tuya: CMD=0x%02X VERSION=%u DATA=[%s] INIT_STATE=%u", command, version, // NOLINT
102  hexencode(message_data, length).c_str(), this->init_state_);
103  this->handle_command_(command, version, message_data, length);
104 
105  // return false to reset rx buffer
106  return false;
107 }

116 void Tuya::handle_command_(uint8_t command, uint8_t version, const uint8_t *buffer, size_t len) {
117  switch ((TuyaCommandType) command) {
118  case TuyaCommandType::HEARTBEAT:
119  ESP_LOGV(TAG, "MCU Heartbeat (0x%02X)", buffer[0]);
120  if (buffer[0] == 0) {
121  ESP_LOGI(TAG, "MCU restarted");
122  this->init_state_ = TuyaInitState::INIT_HEARTBEAT;
123  }
124  if (this->init_state_ == TuyaInitState::INIT_HEARTBEAT) {
125  this->init_state_ = TuyaInitState::INIT_PRODUCT;
126  this->send_empty_command_(TuyaCommandType::PRODUCT_QUERY);
127  }
128  break;
129  case TuyaCommandType::PRODUCT_QUERY: {
130  // check it is a valid string made up of printable characters
131  bool valid = true;
132  for (int i = 0; i < len; i++) {
133  if (!std::isprint(buffer[i])) {
134  valid = false;
135  break;
136  }
137  }
138  if (valid) {
139  this->product_ = std::string(reinterpret_cast<const char *>(buffer), len);
140  } else {
141  this->product_ = R"({"p":"INVALID"})";
142  }
143  if (this->init_state_ == TuyaInitState::INIT_PRODUCT) {
144  this->init_state_ = TuyaInitState::INIT_CONF;
145  this->send_empty_command_(TuyaCommandType::CONF_QUERY);
146  }
147  break;
148  }
149  case TuyaCommandType::CONF_QUERY: {
150  if (len >= 2) {
151  gpio_status_ = buffer[0];
152  gpio_reset_ = buffer[1];
153  }
154  if (this->init_state_ == TuyaInitState::INIT_CONF) {
155  // If mcu returned status gpio, then we can ommit sending wifi state
156  if (this->gpio_status_ != -1) {
157  this->init_state_ = TuyaInitState::INIT_DATAPOINT;
158  this->send_empty_command_(TuyaCommandType::DATAPOINT_QUERY);
159  } else {
160  this->init_state_ = TuyaInitState::INIT_WIFI;
161  // If we were following the spec to the letter we would send
162  // state updates until connected to both WiFi and API/MQTT.
163  // Instead we just claim to be connected immediately and move on.
164  this->send_command_(TuyaCommand{.cmd = TuyaCommandType::WIFI_STATE, .payload = std::vector<uint8_t>{0x04}});
165  }
166  }
167  break;
168  }
169  case TuyaCommandType::WIFI_STATE:
170  if (this->init_state_ == TuyaInitState::INIT_WIFI) {
171  this->init_state_ = TuyaInitState::INIT_DATAPOINT;
172  this->send_empty_command_(TuyaCommandType::DATAPOINT_QUERY);
173  }
174  break;
175  case TuyaCommandType::WIFI_RESET:
176  ESP_LOGE(TAG, "TUYA_CMD_WIFI_RESET is not handled");
177  break;
178  case TuyaCommandType::WIFI_SELECT:
179  ESP_LOGE(TAG, "TUYA_CMD_WIFI_SELECT is not handled");
180  break;
181  case TuyaCommandType::DATAPOINT_DELIVER:
182  break;
183  case TuyaCommandType::DATAPOINT_REPORT:
184  if (this->init_state_ == TuyaInitState::INIT_DATAPOINT) {
185  this->init_state_ = TuyaInitState::INIT_DONE;
186  this->set_timeout("datapoint_dump", 1000, [this] { this->dump_config(); });
187  }
188  this->handle_datapoint_(buffer, len);
189  break;
190  case TuyaCommandType::DATAPOINT_QUERY:
191  break;
192  case TuyaCommandType::WIFI_TEST: {
193  this->send_command_(TuyaCommand{.cmd = TuyaCommandType::WIFI_TEST, .payload = std::vector<uint8_t>{0x00, 0x00}});
194  break;
195  }
196  case TuyaCommandType::LOCAL_TIME_QUERY: {
197 #ifdef USE_TIME
198  if (this->time_id_.has_value()) {
199  auto time_id = *this->time_id_;
200  auto now = time_id->now();
201 
202  if (now.is_valid()) {
203  uint8_t year = now.year - 2000;
204  uint8_t month = now.month;
205  uint8_t day_of_month = now.day_of_month;
206  uint8_t hour = now.hour;
207  uint8_t minute = now.minute;
208  uint8_t second = now.second;
209  // Tuya days starts from Monday, esphome uses Sunday as day 1
210  uint8_t day_of_week = now.day_of_week - 1;
211  if (day_of_week == 0) {
212  day_of_week = 7;
213  }
214  this->send_command_(TuyaCommand{
215  .cmd = TuyaCommandType::LOCAL_TIME_QUERY,
216  .payload = std::vector<uint8_t>{0x01, year, month, day_of_month, hour, minute, second, day_of_week}});
217  } else {
218  ESP_LOGW(TAG, "TUYA_CMD_LOCAL_TIME_QUERY is not handled because time is not valid");
219  // By spec we need to notify MCU that the time was not obtained
220  this->send_command_(
221  TuyaCommand{.cmd = TuyaCommandType::LOCAL_TIME_QUERY,
222  .payload = std::vector<uint8_t>{0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00, 0x00}});
223  }
224  } else {
225  ESP_LOGW(TAG, "TUYA_CMD_LOCAL_TIME_QUERY is not handled because time is not configured");
226  }
227 #else
228  ESP_LOGE(TAG, "LOCAL_TIME_QUERY is not handled");
229 #endif
230  break;
231  }
232  default:
233  ESP_LOGE(TAG, "invalid command (%02x) received", command);
234  }

237 void Tuya::handle_datapoint_(const uint8_t *buffer, size_t len) {
238  if (len < 2)
239  return;
240 
241  TuyaDatapoint datapoint{};
242  datapoint.id = buffer[0];
243  datapoint.type = (TuyaDatapointType) buffer[1];
244  datapoint.value_uint = 0;
245 
246  // drop update if datapoint is in ignore_mcu_datapoint_update list
247  for (auto i : this->ignore_mcu_update_on_datapoints_) {
248  if (datapoint.id == i) {
249  ESP_LOGV(TAG, "Datapoint %u found in ignore_mcu_update_on_datapoints list, dropping MCU update", datapoint.id);
250  return;
251  }
252  }
253 
254  size_t data_size = (buffer[2] << 8) + buffer[3];
255  const uint8_t *data = buffer + 4;
256  size_t data_len = len - 4;
257  if (data_size != data_len) {
258  ESP_LOGW(TAG, "invalid datapoint update");
259  return;
260  }
261 
262  switch (datapoint.type) {
263  case TuyaDatapointType::BOOLEAN:
264  if (data_len != 1)
265  return;
266  datapoint.value_bool = data[0];
267  break;
268  case TuyaDatapointType::INTEGER:
269  if (data_len != 4)
270  return;
271  datapoint.value_uint = encode_uint32(data[0], data[1], data[2], data[3]);
272  break;
273  case TuyaDatapointType::STRING:
274  datapoint.value_string = std::string(reinterpret_cast<const char *>(data), data_len);
275  break;
276  case TuyaDatapointType::ENUM:
277  if (data_len != 1)
278  return;
279  datapoint.value_enum = data[0];
280  break;
281  case TuyaDatapointType::BITMASK:
282  if (data_len != 2)
283  return;
284  datapoint.value_bitmask = (uint16_t(data[0]) << 8) | (uint16_t(data[1]) << 0);
285  break;
286  default:
287  return;
288  }
289  ESP_LOGV(TAG, "Datapoint %u update to %u", datapoint.id, datapoint.value_uint);
290 
291  // Update internal datapoints
292  bool found = false;
293  for (auto &other : this->datapoints_) {
294  if (other.id == datapoint.id) {
295  other = datapoint;
296  found = true;
297  }
298  }
299  if (!found) {
300  this->datapoints_.push_back(datapoint);
301  }
302 
303  // Run through listeners
304  for (auto &listener : this->listeners_)
305  if (listener.datapoint_id == datapoint.id)
306  listener.on_datapoint(datapoint);
307 }

309 void Tuya::send_raw_command_(TuyaCommand command) {
310  uint8_t len_hi = (uint8_t)(command.payload.size() >> 8);
311  uint8_t len_lo = (uint8_t)(command.payload.size() & 0xFF);
312  uint8_t version = 0;
313 
314  this->last_command_timestamp_ = millis();
315 
316  ESP_LOGV(TAG, "Sending Tuya: CMD=0x%02X VERSION=%u DATA=[%s] INIT_STATE=%u", command.cmd, version, // NOLINT
317  hexencode(command.payload).c_str(), this->init_state_);
318 
319  this->write_array({0x55, 0xAA, version, (uint8_t) command.cmd, len_hi, len_lo});
320  if (!command.payload.empty())
321  this->write_array(command.payload.data(), command.payload.size());
322 
323  uint8_t checksum = 0x55 + 0xAA + (uint8_t) command.cmd + len_hi + len_lo;
324  for (auto &data : command.payload)
325  checksum += data;
326  this->write_byte(checksum);
327 }

347 void Tuya::set_datapoint_value(TuyaDatapoint datapoint) {

350  for (auto &other : this->datapoints_) {
351  if (other.id == datapoint.id) {
352  // String value is stored outside the union; must be checked separately.
353  if (datapoint.type == TuyaDatapointType::STRING) {
354  if (other.value_string == datapoint.value_string) {
355  ESP_LOGV(TAG, "Not sending unchanged value");
356  return;
357  }
358  } else if (other.value_uint == datapoint.value_uint) {
359  ESP_LOGV(TAG, "Not sending unchanged value");
360  return;
361  }
362  }
363  }
364  buffer.push_back(datapoint.id);
365  buffer.push_back(static_cast<uint8_t>(datapoint.type));
366 
367  std::vector<uint8_t> data;
368  switch (datapoint.type) {
369  case TuyaDatapointType::BOOLEAN:
370  data.push_back(datapoint.value_bool);
371  break;
372  case TuyaDatapointType::INTEGER:
373  data.push_back(datapoint.value_uint >> 24);
374  data.push_back(datapoint.value_uint >> 16);
375  data.push_back(datapoint.value_uint >> 8);
376  data.push_back(datapoint.value_uint >> 0);
377  break;
378  case TuyaDatapointType::STRING:
379  for (char const &c : datapoint.value_string) {
380  data.push_back(c);
381  }
382  break;
383  case TuyaDatapointType::ENUM:
384  data.push_back(datapoint.value_enum);
385  break;
386  case TuyaDatapointType::BITMASK:
387  data.push_back(datapoint.value_bitmask >> 8);
388  data.push_back(datapoint.value_bitmask >> 0);
389  break;
390  default:
391  return;
392  }
393 
394  buffer.push_back(data.size() >> 8);
395  buffer.push_back(data.size() >> 0);
396  buffer.insert(buffer.end(), data.begin(), data.end());
397 
398  this->send_command_(TuyaCommand{.cmd = TuyaCommandType::DATAPOINT_DELIVER, .payload = buffer});
399 }
