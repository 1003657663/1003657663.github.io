70 void ATM90E32Component::setup() {
71  ESP_LOGCONFIG(TAG, "Setting up ATM90E32 Component...");
72  this->spi_setup();
73 
74  uint16_t mmode0 = 0x87; // 3P4W 50Hz
75  if (line_freq_ == 60) {
76  mmode0 |= 1 << 12; // sets 12th bit to 1, 60Hz
77  }
78 
79  if (current_phases_ == 2) {
80  mmode0 |= 1 << 8; // sets 8th bit to 1, 3P3W
81  mmode0 |= 0 << 1; // sets 1st bit to 0, phase b is not counted into the all-phase sum energy/power (P/Q/S)
82  }
83 
84  this->write16_(ATM90E32_REGISTER_SOFTRESET, 0x789A); // Perform soft reset
85  this->write16_(ATM90E32_REGISTER_CFGREGACCEN, 0x55AA); // enable register config access
86  this->write16_(ATM90E32_REGISTER_METEREN, 0x0001); // Enable Metering
87  if (this->read16_(ATM90E32_REGISTER_LASTSPIDATA) != 0x0001) {
88  ESP_LOGW(TAG, "Could not initialize ATM90E32 IC, check SPI settings");
89  this->mark_failed();
90  return;
91  }
92  this->write16_(ATM90E32_REGISTER_PLCONSTH, 0x0861); // PL Constant MSB (default) = 140625000
93  this->write16_(ATM90E32_REGISTER_PLCONSTL, 0xC468); // PL Constant LSB (default)
94  this->write16_(ATM90E32_REGISTER_ZXCONFIG, 0xD654); // ZX2, ZX1, ZX0 pin config
95  this->write16_(ATM90E32_REGISTER_MMODE0, mmode0); // Mode Config (frequency set in main program)
96  this->write16_(ATM90E32_REGISTER_MMODE1, pga_gain_); // PGA Gain Configuration for Current Channels
97  this->write16_(ATM90E32_REGISTER_PSTARTTH, 0x1D4C); // All Active Startup Power Threshold - 0.02A/0.00032 = 7500
98  this->write16_(ATM90E32_REGISTER_QSTARTTH, 0x1D4C); // All Reactive Startup Power Threshold - 50%
99  this->write16_(ATM90E32_REGISTER_PPHASETH, 0x02EE); // Each Phase Active Phase Threshold - 0.002A/0.00032 = 750
100  this->write16_(ATM90E32_REGISTER_QPHASETH, 0x02EE); // Each phase Reactive Phase Threshold - 10%
101  this->write16_(ATM90E32_REGISTER_UGAINA, this->phase_[0].volt_gain_); // A Voltage rms gain
102  this->write16_(ATM90E32_REGISTER_IGAINA, this->phase_[0].ct_gain_); // A line current gain
103  this->write16_(ATM90E32_REGISTER_UGAINB, this->phase_[1].volt_gain_); // B Voltage rms gain
104  this->write16_(ATM90E32_REGISTER_IGAINB, this->phase_[1].ct_gain_); // B line current gain
105  this->write16_(ATM90E32_REGISTER_UGAINC, this->phase_[2].volt_gain_); // C Voltage rms gain
106  this->write16_(ATM90E32_REGISTER_IGAINC, this->phase_[2].ct_gain_); // C line current gain
107  this->write16_(ATM90E32_REGISTER_CFGREGACCEN, 0x0000); // end configuration
108 }
