61 void APIServer::loop() {
62  // Partition clients into remove and active
63  auto new_end =
64  std::partition(this->clients_.begin(), this->clients_.end(), [](APIConnection *conn) { return !conn->remove_; });
65  // print disconnection messages
66  for (auto it = new_end; it != this->clients_.end(); ++it) {
67  ESP_LOGD(TAG, "Disconnecting %s", (*it)->client_info_.c_str());
68  }
69  // only then delete the pointers, otherwise log routine
70  // would access freed memory
71  for (auto it = new_end; it != this->clients_.end(); ++it)
72  delete *it;
73  // resize vector
74  this->clients_.erase(new_end, this->clients_.end());
75 
76  for (auto *client : this->clients_) {
77  client->loop();
78  }
79 
80  if (this->reboot_timeout_ != 0) {
81  const uint32_t now = millis();
82  if (!this->is_connected()) {
83  if (now - this->last_connected_ > this->reboot_timeout_) {
84  ESP_LOGE(TAG, "No client connected to API. Rebooting...");
85  App.reboot();
86  }
87  this->status_set_warning();
88  } else {
89  this->last_connected_ = now;
90  this->status_clear_warning();
91  }
92  }

114  result = 0;

118  result = 1;
