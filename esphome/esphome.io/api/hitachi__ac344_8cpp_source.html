8 void set_bits(uint8_t *const dst, const uint8_t offset, const uint8_t nbits, const uint8_t data) {
9  if (offset >= 8 || !nbits)
10  return; // Short circuit as it won't change.
11  // Calculate the mask for the supplied value.
12  uint8_t mask = UINT8_MAX >> (8 - ((nbits > 8) ? 8 : nbits));
13  // Calculate the mask & clear the space for the data.
14  // Clear the destination bits.
15  *dst &= ~(uint8_t)(mask << offset);
16  // Merge in the data.
17  *dst |= ((data & mask) << offset);
18 }

311 bool HitachiClimate::on_receive(remote_base::RemoteReceiveData data) {
312  // Validate header
313  if (!data.expect_item(HITACHI_AC344_HDR_MARK, HITACHI_AC344_HDR_SPACE)) {
314  ESP_LOGVV(TAG, "Header fail");
315  return false;
316  }
317 
318  uint8_t recv_state[HITACHI_AC344_STATE_LENGTH] = {0};
319  // Read all bytes.
320  for (uint8_t pos = 0; pos < HITACHI_AC344_STATE_LENGTH; pos++) {
321  // Read bit
322  for (int8_t bit = 0; bit < 8; bit++) {
323  if (data.expect_item(HITACHI_AC344_BIT_MARK, HITACHI_AC344_ONE_SPACE))
324  recv_state[pos] |= 1 << bit;
325  else if (!data.expect_item(HITACHI_AC344_BIT_MARK, HITACHI_AC344_ZERO_SPACE)) {
326  ESP_LOGVV(TAG, "Byte %d bit %d fail", pos, bit);
327  return false;
328  }
329  }
330  }
331 
332  // Validate footer
333  if (!data.expect_mark(HITACHI_AC344_BIT_MARK)) {
334  ESP_LOGVV(TAG, "Footer fail");
335  return false;
336  }
337 
338  dump_state_("Recv", recv_state);
339 
340  // parse mode
341  this->parse_mode_(recv_state);
342  // parse temperature
343  this->parse_temperature_(recv_state);
344  // parse fan
345  this->parse_fan_(recv_state);
346  // parse swingv
347  this->parse_swing_(recv_state);
348  this->publish_state();
349  for (uint8_t i = 0; i < HITACHI_AC344_STATE_LENGTH; i++)
350  remote_state_[i] = recv_state[i];
351 
352  return true;
353 }
