10 //
11 // According to the datasheet, the component is supposed to respond in more than 75ms. In fact, it can answer almost
12 // immediately for temperature. But for humidity, it takes >90ms to get a valid data. From experience, we have best
13 // results making successive requests; the current implementation make 3 attemps with a delay of 30ms each time.

31  if (!this->write_bytes(0, AHT10_CALIBRATE_CMD, sizeof(AHT10_CALIBRATE_CMD))) {
32  ESP_LOGE(TAG, "Communication with AHT10 failed!");
33  this->mark_failed();

35  }
36  uint8_t data;
37  if (!this->read_byte(0, &data, AHT10_DEFAULT_DELAY)) {

51 void AHT10Component::update() {
52  if (!this->write_bytes(0, AHT10_MEASURE_CMD, sizeof(AHT10_MEASURE_CMD))) {
53  ESP_LOGE(TAG, "Communication with AHT10 failed!");
54  this->status_set_warning();
55  return;
56  }
57  uint8_t data[6];
58  uint8_t delay = AHT10_DEFAULT_DELAY;
59  if (this->humidity_sensor_ != nullptr)
60  delay = AHT10_HUMIDITY_DELAY;
61  for (int i = 0; i < AHT10_ATTEMPS; ++i) {
62  ESP_LOGVV(TAG, "Attemps %u at %6ld", i, millis());
63  if (!this->read_bytes(0, data, 6, delay)) {
64  ESP_LOGD(TAG, "Communication with AHT10 failed, waiting...");
65  } else if ((data[0] & 0x80) == 0x80) { // Bit[7] = 0b1, device is busy
66  ESP_LOGD(TAG, "AHT10 is busy, waiting...");
67  } else if (data[1] == 0x0 && data[2] == 0x0 && (data[3] >> 4) == 0x0) {
68  // Unrealistic humidity (0x0)
69  if (this->humidity_sensor_ == nullptr) {
70  ESP_LOGVV(TAG, "ATH10 Unrealistic humidity (0x0), but humidity is not required");
71  break;
72  } else {
73  ESP_LOGD(TAG, "ATH10 Unrealistic humidity (0x0), retrying...");
74  if (!this->write_bytes(0, AHT10_MEASURE_CMD, sizeof(AHT10_MEASURE_CMD))) {
75  ESP_LOGE(TAG, "Communication with AHT10 failed!");
76  this->status_set_warning();
77  return;
78  }
79  }
80  } else {
81  // data is valid, we can break the loop
82  ESP_LOGVV(TAG, "Answer at %6ld", millis());
83  break;
84  }
85  }
86  if ((data[0] & 0x80) == 0x80) {
87  ESP_LOGE(TAG, "Measurements reading timed-out!");
88  this->status_set_warning();
89  return;
90  }
91 
92  uint32_t raw_temperature = ((data[3] & 0x0F) << 16) | (data[4] << 8) | data[5];
93  uint32_t raw_humidity = ((data[1] << 16) | (data[2] << 8) | data[3]) >> 4;
94 
95  float temperature = ((200.0 * (float) raw_temperature) / 1048576.0) - 50.0;
96  float humidity;
97  if (raw_humidity == 0) { // unrealistic value
98  humidity = NAN;
99  } else {
100  humidity = (float) raw_humidity * 100.0 / 1048576.0;
101  }
102 
103  if (this->temperature_sensor_ != nullptr) {
104  this->temperature_sensor_->publish_state(temperature);
105  }
106  if (this->humidity_sensor_ != nullptr) {
107  if (isnan(humidity))
108  ESP_LOGW(TAG, "Invalid humidity! Sensor reported 0%% Hum");
109  this->humidity_sensor_->publish_state(humidity);
110  }
111  this->status_clear_warning();
112 }
