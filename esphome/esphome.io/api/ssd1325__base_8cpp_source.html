192 void HOT SSD1325::draw_absolute_pixel_internal(int x, int y, Color color) {
193  if (x >= this->get_width_internal() || x < 0 || y >= this->get_height_internal() || y < 0)
194  return;
195  uint32_t color4 = display::ColorUtil::color_to_grayscale4(color);
196  // where should the bits go in the big buffer array? math...
197  uint16_t pos = (x / SSD1325_PIXELSPERBYTE) + (y * this->get_width_internal() / SSD1325_PIXELSPERBYTE);
198  uint8_t shift = (x % SSD1325_PIXELSPERBYTE) * SSD1325_COLORSHIFT;
199  // ensure 'color4' is valid (only 4 bits aka 1 nibble) and shift the bits left when necessary
200  color4 = (color4 & SSD1325_COLORMASK) << shift;
201  // first mask off the nibble we must change...
202  this->buffer_[pos] &= (~SSD1325_COLORMASK >> shift);
203  // ...then lay the new nibble back on top. done!
204  this->buffer_[pos] |= color4;
205 }
