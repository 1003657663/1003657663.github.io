153 void MCP2515::prepare_id_(uint8_t *buffer, const bool extended, const uint32_t id) {
154  uint16_t canid = (uint16_t)(id & 0x0FFFF);
155 
156  if (extended) {
157  buffer[MCP_EID0] = (uint8_t)(canid & 0xFF);
158  buffer[MCP_EID8] = (uint8_t)(canid >> 8);
159  canid = (uint16_t)(id >> 16);
160  buffer[MCP_SIDL] = (uint8_t)(canid & 0x03);
161  buffer[MCP_SIDL] += (uint8_t)((canid & 0x1C) << 3);
162  buffer[MCP_SIDL] |= TXB_EXIDE_MASK;
163  buffer[MCP_SIDH] = (uint8_t)(canid >> 5);
164  } else {
165  buffer[MCP_SIDH] = (uint8_t)(canid >> 3);
166  buffer[MCP_SIDL] = (uint8_t)((canid & 0x07) << 5);
167  buffer[MCP_EID0] = 0;
168  buffer[MCP_EID8] = 0;
169  }
170 }
