121 void SGP30Component::send_env_data_() {
122  if (this->humidity_sensor_ == nullptr && this->temperature_sensor_ == nullptr)
123  return;
124  float humidity = NAN;
125  if (this->humidity_sensor_ != nullptr)
126  humidity = this->humidity_sensor_->state;
127  if (isnan(humidity) || humidity < 0.0f || humidity > 100.0f) {
128  ESP_LOGW(TAG, "Compensation not possible yet: bad humidity data.");
129  return;
130  } else {
131  ESP_LOGD(TAG, "External compensation data received: Humidity %0.2f%%", humidity);
132  }
133  float temperature = NAN;
134  if (this->temperature_sensor_ != nullptr) {
135  temperature = float(this->temperature_sensor_->state);
136  }
137  if (isnan(temperature) || temperature < -40.0f || temperature > 85.0f) {
138  ESP_LOGW(TAG, "Compensation not possible yet: bad temperature value data.");
139  return;
140  } else {
141  ESP_LOGD(TAG, "External compensation data received: Temperature %0.2f°C", temperature);
142  }
143 
144  float absolute_humidity =
145  216.7f * (((humidity / 100) * 6.112f * std::exp((17.62f * temperature) / (243.12f + temperature))) /
146  (273.15f + temperature));
147  uint8_t humidity_full = uint8_t(std::floor(absolute_humidity));
148  uint8_t humidity_dec = uint8_t(std::floor((absolute_humidity - std::floor(absolute_humidity)) * 256));
149  ESP_LOGD(TAG, "Calculated Absolute humidity: %0.3f g/m³ (0x%04X)", absolute_humidity,
150  uint16_t(uint16_t(humidity_full) << 8 | uint16_t(humidity_dec)));
151  uint8_t crc = sht_crc_(humidity_full, humidity_dec);
152  uint8_t data[4];
153  data[0] = SGP30_CMD_SET_ABSOLUTE_HUMIDITY & 0xFF;
154  data[1] = humidity_full;
155  data[2] = humidity_dec;
156  data[3] = crc;
157  if (!this->write_bytes(SGP30_CMD_SET_ABSOLUTE_HUMIDITY >> 8, data, 4)) {
158  ESP_LOGE(TAG, "Error sending compensation data.");
159  }
160 }

162 void SGP30Component::write_iaq_baseline_(uint16_t eco2_baseline, uint16_t tvoc_baseline) {
163  uint8_t data[7];
164  data[0] = SGP30_CMD_SET_IAQ_BASELINE & 0xFF;
165  data[1] = tvoc_baseline >> 8;
166  data[2] = tvoc_baseline & 0xFF;
167  data[3] = sht_crc_(data[1], data[2]);
168  data[4] = eco2_baseline >> 8;
169  data[5] = eco2_baseline & 0xFF;
170  data[6] = sht_crc_(data[4], data[5]);
171  if (!this->write_bytes(SGP30_CMD_SET_IAQ_BASELINE >> 8, data, 7)) {
172  ESP_LOGE(TAG, "Error applying eCO2 baseline: 0x%04X, TVOC baseline: 0x%04X", eco2_baseline, tvoc_baseline);
173  } else
174  ESP_LOGI(TAG, "Initial eCO2 and TVOC baselines applied successfully!");

253 uint8_t SGP30Component::sht_crc_(uint8_t data1, uint8_t data2) {
254  uint8_t bit;
255  uint8_t crc = 0xFF;
256 
257  crc ^= data1;
258  for (bit = 8; bit > 0; --bit) {
259  if (crc & 0x80)
260  crc = (crc << 1) ^ 0x131;
261  else
262  crc = (crc << 1);
263  }
264 
265  crc ^= data2;
266  for (bit = 8; bit > 0; --bit) {
267  if (crc & 0x80)
268  crc = (crc << 1) ^ 0x131;
269  else
270  crc = (crc << 1);
271  }
272 
273  return crc;

276 bool SGP30Component::read_data_(uint16_t *data, uint8_t len) {
277  const uint8_t num_bytes = len * 3;
278  auto *buf = new uint8_t[num_bytes];
279 
280  if (!this->parent_->raw_receive(this->address_, buf, num_bytes)) {
281  delete[](buf);
282  return false;
283  }
284 
285  for (uint8_t i = 0; i < len; i++) {
286  const uint8_t j = 3 * i;
287  uint8_t crc = sht_crc_(buf[j], buf[j + 1]);
288  if (crc != buf[j + 2]) {
289  ESP_LOGE(TAG, "CRC8 Checksum invalid! 0x%02X != 0x%02X", buf[j + 2], crc);
290  delete[](buf);
291  return false;
292  }
293  data[i] = (buf[j] << 8) | buf[j + 1];
294  }
295 
296  delete[](buf);
297  return true;
298 }
