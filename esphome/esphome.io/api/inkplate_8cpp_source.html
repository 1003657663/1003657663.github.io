545 void Inkplate6::clean_fast_(uint8_t c, uint8_t rep) {
546  ESP_LOGV(TAG, "Clean fast called with: (%d, %d)", c, rep);
547  unsigned long start_time = millis();
548 
549  eink_on_();
550  uint8_t data = 0;
551  if (c == 0) // White
552  data = B10101010;
553  else if (c == 1) // Black
554  data = B01010101;
555  else if (c == 2) // Discharge
556  data = B00000000;
557  else if (c == 3) // Skip
558  data = B11111111;
559 
560  uint32_t send = ((data & B00000011) << 4) | (((data & B00001100) >> 2) << 18) | (((data & B00010000) >> 4) << 23) |
561  (((data & B11100000) >> 5) << 25);
562  uint32_t clock = (1 << this->cl_pin_->get_pin());
563 
564  for (int k = 0; k < rep; k++) {
565  vscan_start_();
566  for (int i = 0; i < this->get_height_internal(); i++) {
567  hscan_start_(send);
568  GPIO.out_w1ts = send | clock;
569  GPIO.out_w1tc = clock;
570  for (int j = 0, jm = this->get_width_internal() / 8; j < jm; j++) {
571  GPIO.out_w1ts = clock;
572  GPIO.out_w1tc = clock;
573  GPIO.out_w1ts = clock;
574  GPIO.out_w1tc = clock;
575  }
576  GPIO.out_w1ts = clock;
577  GPIO.out_w1tc = get_data_pin_mask_() | clock;
578  vscan_end_();
579  }
580  delayMicroseconds(230);
581  ESP_LOGV(TAG, "Clean fast rep loop %d finished (%lums)", k, millis() - start_time);
582  }
