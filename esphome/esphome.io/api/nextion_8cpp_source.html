112 bool Nextion::read_until_ack_() {
113  while (this->available() >= 4) {
114  // flush preceding filler bytes
115  uint8_t temp;
116  while (this->available() && this->peek_byte(&temp) && temp == 0xFF)
117  this->read_byte(&temp);
118 
119  if (!this->available())
120  break;
121 
122  uint8_t event;
123  // event type
124  this->read_byte(&event);
125 
126  uint8_t data[255];
127  // total length of data (including end bytes)
128  uint8_t data_length = 0;
129  // message is terminated by three consecutive 0xFF
130  // this variable keeps track of ohow many of those have
131  // been received
132  uint8_t end_length = 0;
133  while (this->available() && end_length < 3 && data_length < sizeof(data)) {
134  uint8_t byte;
135  this->read_byte(&byte);
136  if (byte == 0xFF) {
137  end_length++;
138  } else {
139  end_length = 0;
140  }
141  data[data_length++] = byte;
142  }
143 
144  if (end_length != 3) {
145  ESP_LOGW(TAG, "Received unknown filler end bytes from Nextion!");
146  continue;
147  }
148 
149  data_length -= 3; // remove filler bytes
150 
151  bool invalid_data_length = false;
152  switch (event) {
153  case 0x01: // successful execution of instruction (ACK)
154  return true;
155  case 0x00: // invalid instruction
156  ESP_LOGW(TAG, "Nextion reported invalid instruction!");
157  break;
158  case 0x02: // component ID invalid
159  ESP_LOGW(TAG, "Nextion reported component ID invalid!");
160  break;
161  case 0x03: // page ID invalid
162  ESP_LOGW(TAG, "Nextion reported page ID invalid!");
163  break;
164  case 0x04: // picture ID invalid
165  ESP_LOGW(TAG, "Nextion reported picture ID invalid!");
166  break;
167  case 0x05: // font ID invalid
168  ESP_LOGW(TAG, "Nextion reported font ID invalid!");
169  break;
170  case 0x11: // baud rate setting invalid
171  ESP_LOGW(TAG, "Nextion reported baud rate invalid!");
172  break;
173  case 0x12: // curve control ID number or channel number is invalid
174  ESP_LOGW(TAG, "Nextion reported control/channel ID invalid!");
175  break;
176  case 0x1A: // variable name invalid
177  ESP_LOGW(TAG, "Nextion reported variable name invalid!");
178  break;
179  case 0x1B: // variable operation invalid
180  ESP_LOGW(TAG, "Nextion reported variable operation invalid!");
181  break;
182  case 0x1C: // failed to assign
183  ESP_LOGW(TAG, "Nextion reported failed to assign variable!");
184  break;
185  case 0x1D: // operate EEPROM failed
186  ESP_LOGW(TAG, "Nextion reported operating EEPROM failed!");
187  break;
188  case 0x1E: // parameter quantity invalid
189  ESP_LOGW(TAG, "Nextion reported parameter quantity invalid!");
190  break;
191  case 0x1F: // IO operation failed
192  ESP_LOGW(TAG, "Nextion reported component I/O operation invalid!");
193  break;
194  case 0x20: // undefined escape characters
195  ESP_LOGW(TAG, "Nextion reported undefined escape characters!");
196  break;
197  case 0x23: // too long variable name
198  ESP_LOGW(TAG, "Nextion reported too long variable name!");
199  break;
200  case 0x65: { // touch event return data
201  if (data_length != 3) {
202  invalid_data_length = true;
203  break;
204  }
205  uint8_t page_id = data[0];
206  uint8_t component_id = data[1];
207  uint8_t touch_event = data[2]; // 0 -> release, 1 -> press
208  ESP_LOGD(TAG, "Got touch page=%u component=%u type=%s", page_id, component_id,
209  touch_event ? "PRESS" : "RELEASE");
210  for (auto *touch : this->touch_) {
211  touch->process(page_id, component_id, touch_event);
212  }
213  break;
214  }
215  case 0x67:
216  case 0x68: { // touch coordinate data
217  if (data_length != 5) {
218  invalid_data_length = true;
219  break;
220  }
221  uint16_t x = (uint16_t(data[0]) << 8) | data[1];
222  uint16_t y = (uint16_t(data[2]) << 8) | data[3];
223  uint8_t touch_event = data[4]; // 0 -> release, 1 -> press
224  ESP_LOGD(TAG, "Got touch at x=%u y=%u type=%s", x, y, touch_event ? "PRESS" : "RELEASE");
225  break;
226  }
227  case 0x66: // sendme page id
228  case 0x70: // string variable data return
229  case 0x71: // numeric variable data return
230  case 0x86: // device automatically enters into sleep mode
231  case 0x87: // device automatically wakes up
232  case 0x88: // system successful start up
233  case 0x89: // start SD card upgrade
234  case 0xFD: // data transparent transmit finished
235  case 0xFE: // data transparent transmit ready
236  break;
237  default:
238  ESP_LOGW(TAG, "Received unknown event from nextion: 0x%02X", event);
239  break;
240  }
241  if (invalid_data_length) {
242  ESP_LOGW(TAG, "Invalid data length from nextion!");
243  }
244  }
245 
246  return false;
