138  void normalize_color(const LightTraits &traits) {
139  if (traits.get_supports_rgb()) {
140  float max_value = fmaxf(this->get_red(), fmaxf(this->get_green(), this->get_blue()));
141  if (traits.get_supports_rgb_white_value()) {
142  max_value = fmaxf(max_value, this->get_white());
143  if (max_value == 0.0f) {
144  this->set_white(1.0f);
145  } else {
146  this->set_white(this->get_white() / max_value);
147  }
148  }
149  if (max_value == 0.0f) {
150  this->set_red(1.0f);
151  this->set_green(1.0f);
152  this->set_blue(1.0f);
153  } else {
154  this->set_red(this->get_red() / max_value);
155  this->set_green(this->get_green() / max_value);
156  this->set_blue(this->get_blue() / max_value);
157  }
158  }
159 
160  if (traits.get_supports_brightness() && this->get_brightness() == 0.0f) {
161  if (traits.get_supports_rgb_white_value()) {
162  // 0% brightness for RGBW[W] means no RGB channel, but white channel on.
163  // do nothing
164  } else {
165  // 0% brightness means off
166  this->set_state(false);
167  // reset brightness to 100%

200  void as_rgbww(float color_temperature_cw, float color_temperature_ww, float *red, float *green, float *blue,
201  float *cold_white, float *warm_white, float gamma = 0, bool constant_brightness = false,
