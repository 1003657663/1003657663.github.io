157  this->mode = climate::CLIMATE_MODE_DRY;

208  uint8_t state_frame[DAIKIN_STATE_FRAME_SIZE] = {};
209  if (!data.expect_item(DAIKIN_HEADER_MARK, DAIKIN_HEADER_SPACE)) {
210  return false;

213  uint8_t byte = 0;

218  return false;

220  }

222  if (pos == 0) {

224  if (byte != 0x11)
225  return false;
226  } else if (pos == 1) {

228  if (byte != 0xDA)
229  return false;

233  return false;

237  return false;
